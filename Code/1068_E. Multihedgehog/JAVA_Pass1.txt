import java.io.*;
import java.util.*;

public class E implements Runnable {
	public static void main (String[] args) {new Thread(null, new E(), "_cf", 1 << 28).start();}

	public void run() {
		FastScanner fs = new FastScanner();
		PrintWriter out = new PrintWriter(System.out);
		System.err.println("");

		int n = fs.nextInt();
		int k = fs.nextInt();
		ArrayList<Integer>[] adj = new ArrayList[n];
		for(int i = 0; i < n; i++) adj[i] = new ArrayList<Integer>();
		int[] deg = new int[n];
		for(int i = 0; i < n-1; i++) {
			int u = fs.nextInt()-1;
			int v = fs.nextInt()-1;
			adj[u].add(v);
			adj[v].add(u);
			deg[v]++; deg[u]++;
		}

		int[] odeg = deg.clone();
		ArrayDeque<Integer> dq = new ArrayDeque<Integer>();
		int[] time = new int[n];
		Arrays.fill(time, -1);
		boolean[] clear = new boolean[n];
		for(int i = 0; i < n; i++) if(deg[i] == 1) {
			time[i] = 0;
			dq.add(i);
			clear[i] = true;
		}
		
		int center = -1;
		while(!dq.isEmpty()) {
			int u = dq.poll();
			center = u;
			for(int v : adj[u]) {
				if(clear[v]) continue;
				deg[v]--;
				if(deg[v] == 1) {
					time[v] = time[u]+1;
					clear[v] = true;
					dq.add(v);
				}
				if(deg[v] == 0) {
					center = v;
					break;
				}
			}
		}
		if(center == -1) {
			System.out.println("No");
			return;
		}
		
		deg = odeg;
//		System.out.println((center+1));
		if(deg[center] < 3) {
			System.out.println("No");
			return;
		}

		time = new int[n];
		time[center] = k;
		dq.add(center);
		clear = new boolean[n];
		clear[center] = true;
		while(!dq.isEmpty()) {
			int u = dq.poll();
			clear[u] = true;
			if(time[u] > 0 && deg[u] < 3) {
				System.out.println("No");
				return;
			}
			if(time[u] == 0 && deg[u] != 0) {
				System.out.println("No");
				return;
			}
			
			for(int v : adj[u]) {
				if(clear[v]) continue;
				deg[v]--;
				time[v] = time[u]-1;
				clear[v] = true;
				dq.add(v);
			}
		}
		
		out.println("Yes");
		
		out.close();
	}

	class FastScanner {
		public int BS = 1<<16;
		public char NC = (char)0;
		byte[] buf = new byte[BS];
		int bId = 0, size = 0;
		char c = NC;
		double num = 1;
		BufferedInputStream in;

		public FastScanner() {
			in = new BufferedInputStream(System.in, BS);
		}

		public FastScanner(String s) {
			try {
				in = new BufferedInputStream(new FileInputStream(new File(s)), BS);
			}
			catch (Exception e) {
				in = new BufferedInputStream(System.in, BS);
			}
		}

		public char nextChar(){
			while(bId==size) {
				try {
					size = in.read(buf);
				}catch(Exception e) {
					return NC;
				}                
				if(size==-1)return NC;
				bId=0;
			}
			return (char)buf[bId++];
		}

		public int nextInt() {
			return (int)nextLong();
		}

		public long nextLong() {
			num=1;
			boolean neg = false;
			if(c==NC)c=nextChar();
			for(;(c<'0' || c>'9'); c = nextChar()) {
				if(c=='-')neg=true;
			}
			long res = 0;
			for(; c>='0' && c <='9'; c=nextChar()) {
				res = (res<<3)+(res<<1)+c-'0';
				num*=10;
			}
			return neg?-res:res;
		}

		public double nextDouble() {
			double cur = nextLong();
			return c!='.' ? cur:cur+nextLong()/num;
		}

		public String next() {
			StringBuilder res = new StringBuilder();
			while(c<=32)c=nextChar();
			while(c>32) {
				res.append(c);
				c=nextChar();
			}
			return res.toString();
		}

		public String nextLine() {
			StringBuilder res = new StringBuilder();
			while(c<=32)c=nextChar();
			while(c!='\n') {
				res.append(c);
				c=nextChar();
			}
			return res.toString();
		}

		public boolean hasNext() {
			if(c>32)return true;
			while(true) {
				c=nextChar();
				if(c==NC)return false;
				else if(c>32)return true;
			}
		}
		
		public int[] nextIntArray(int n) {
			int[] res = new int[n];
			for(int i = 0; i < n; i++) res[i] = nextInt();
			return res;
		}
		
	}

}
