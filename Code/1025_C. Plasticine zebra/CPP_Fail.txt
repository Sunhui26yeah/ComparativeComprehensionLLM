#include <bits/stdc++.h>
using namespace std;
int main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);
  string s;
  cin >> s;
  long long ans = 0;
  long long cnt = 1;
  for (int i = 1; i < s.size(); i++) {
    if (s[i] != s[i - 1])
      cnt++;
    else {
      ans = max(cnt, ans);
      cnt = 1;
    }
  }
  if (s[0] != s[s.size() - 1]) {
    int i = 0;
    int j = s.size() - 1;
    cnt = 0;
    while (i < s.size() && s[i] != s[i + 1]) {
      cnt++;
      i++;
    }
    cnt++;
    while (j >= 0 && s[j] != s[j - 1]) {
      cnt++;
      j--;
    }
    cnt++;
    ans = max(ans, cnt);
  }
  cout << ans;
  return 0;
}
