s=input()
n=len(s)
maxa=-1
c=1
for i in range(0,n-1):
    if s[i]!=s[i+1]:
        c+=1
    else:
        maxa=max(maxa,c)
        c=1
c=1
if maxa==1:
    print(1)
    exit()
for i in range(n-1,0,-1):
    if s[i]!=s[i-1]:
        c+=1
    else:
        break
co=1
for i in range(0,n-1):
    if s[i]!=s[i+1]:
        co+=1
    elif s[i]==s[i+1]:
        if s[i]==s[n-1]:
            maxa=max(co+c,maxa)
            print(maxa)
            exit()
print(maxa)
