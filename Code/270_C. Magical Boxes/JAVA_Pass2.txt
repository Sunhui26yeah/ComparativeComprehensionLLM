import java.util.*;
import java.io.*;
public class MagicalBoxes
{
	public static void main(String[] args)
	{
		Scanner in = new Scanner(System.in);
		
		int n = in.nextInt();
		
		Box[] boxes = new Box[n];
		for(int x = 0; x < n; x++)
		{
			boxes[x] = new Box(in.nextInt(), in.nextInt());
		}
		
		Arrays.sort(boxes);
		
		for(int y = 0; y < boxes.length - 1; y++)
		{
			boxes[y + 1].num = Math.max(boxes[y + 1].num, (boxes[y].num - 1) / (long)Math.pow(4, boxes[y + 1].size - boxes[y].size) + 1);
		}
		
		int z;
		for(z = 1; z < 100; z++)
		{
			if(boxes[n - 1].num <= (long)Math.pow(4, z))
			{
				break;
			}
		}
		
		System.out.println((boxes[n - 1].size + z));
	}
}

class Box implements Comparable<Box>
{
	long size;
	long num;
	
	public Box(int s, int n)
	{
		size = s;
		num = n;
	}
	
	public int compareTo(Box o)
	{
		if(size < o.size)
		{
			return -1;
		}
		else if(size > o.size)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}
}
