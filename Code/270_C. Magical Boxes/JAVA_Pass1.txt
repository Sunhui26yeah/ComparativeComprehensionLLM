import java.util.*;

public class Main{
	public static void main(String args[]){
		new Main().run();
	}

	public void run(){
		Scanner scanner = new Scanner(System.in);
		String line = scanner.nextLine();
		int n = Integer.parseInt(line);
		double[][] ka = new double[n][2];
		for(int i = 0; i < n; i++){
			line = scanner.nextLine();
			for(int j = 0; j < 2; j++){
				ka[i][j] = Integer.parseInt(line.split(" ")[j]);
				//System.out.print(ka[i][j] + " ");
			}
			//System.out.println();
		}
		//System.out.println();
		
		//init
		int res = 0;
		for(int i = 0; i < n; i++){
			int tmpRes = 0;
			if(ka[i][1] > 0){
				tmpRes = (int)ka[i][0] + 1;
				//System.out.println("i:" + i + " ka[i][1]:" + ka[i][1] + " ka[i][1]/4" + ka[i][1]/4);
			}
			while(ka[i][1] / 4 > 1) {
				ka[i][1] /= 4;
				tmpRes++;
				//System.out.println("i:" + i + " ka[i][1]:" + ka[i][1]);
			}
			
			res = Math.max(res, tmpRes);
		}
		System.out.println(res);

		
	}
}
