n,u=map(int,input().split())
arr=list(map(int,input().split()))
arr.sort()
j,i=1,0
maxi=-1
while(i<n-2):
	while(1):
		if j>=n:
			j=n-1
			break
		if arr[j]-arr[i]>u:
			j-=1
			break
		j+=1
	if i==j:
		j+=1
	elif arr[j]==arr[i]:
		pass
	elif arr[j]-arr[i]<=u:
		# print(i,j)
		maxi=max(maxi,(arr[j]-arr[i+1])/(arr[j]-arr[i]))
	i+=1
print(maxi)
