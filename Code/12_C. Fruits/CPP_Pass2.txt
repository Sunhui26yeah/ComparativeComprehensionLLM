vector<int> price;
map<string, int> MP;
struct node {
  string s;
  int n;
};
bool comp(node a, node b) { return a.n > b.n; }
int main() {
  int n, m;
  cin >> n >> m;
  for (long long i = 0; i < n; i++) {
    int temp;
    cin >> temp;
    price.push_back(temp);
  }
  sort(price.begin(), price.end());
  vector<int> VV;
  VV = price;
  for (int i = 0; i < m; i++) {
    string str;
    cin >> str;
    MP[str]++;
  }
  vector<node> V;
  map<string, int>::iterator it;
  for (it = MP.begin(); it != MP.end(); it++) {
    node temp;
    string first;
    int second;
    first = (it->first);
    second = it->second;
    temp.s = first;
    temp.n = second;
    V.push_back(temp);
  }
  sort(V.begin(), V.end(), comp);
  int maxi = 0;
  for (int i = 0; i < V.size(); i++) {
    maxi += (V[i].n * VV.back());
    VV.pop_back();
  }
  reverse(price.begin(), price.end());
  int mini = 0;
  for (int i = 0; i < V.size(); i++) {
    mini += (V[i].n * price.back());
    price.pop_back();
  }
  cout << mini << " " << maxi << "\n";
  return 0;
}
