import java.util.Scanner;

public class EasyGame {
	private static char[][] board = new char[3][3];
	private static int numX = 0;
	private static int num0 = 0;

	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		for (int i = 0; i < 3; i++) {
			String line = scanner.nextLine();
			for (int j = 0; j < 3; j++) {
				board[i][j] = line.charAt(j);
				if (board[i][j] == 'X') {
					numX++;
				} else if (board[i][j] == '0') {
					num0++;
				}
			}
		}

		if (!isLegal()) {
			System.out.println("illegal");
		} else if (hasWon('X')) {
			System.out.println("the first player won\n");
		} else if (hasWon('0')) {
			System.out.println("the second player won\n");
		} else if (numX + num0 == 9) {
			System.out.println("draw\n");
		} else if (numX == num0) {
			System.out.println("first\n");
		} else if (numX - num0 == 1) {
			System.out.println("second\n");
		}
	}

	private static boolean isLegal () {
		return numX - num0 <= 1 && numX - num0 >= 0 && !(hasWon('X') && numX == num0) && !(hasWon('0') && numX - num0 == 1);
	}

	private static boolean hasWon (char c) {
		for (int i = 0; i < 3; i++) {
			if (board[i][0] == c && board[i][1] == c && board[i][2] == c) {
				return true;
			}
			if (board[0][i] == c && board[1][i] == c && board[2][i] == c) {
				return true;
			}
		}

		return board[1][1] == c && (board[0][0] == c && board[2][2] == c || board[0][2] == c && board[2][0] == c);
	}
}
