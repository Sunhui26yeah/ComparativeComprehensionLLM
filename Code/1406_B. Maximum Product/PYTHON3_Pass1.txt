for t in range(int(input())):
    n=int(input())
    a=list(map(int, input().split()))
    cp=[]
    cn=[]
    for i in range(n):
        if a[i]<0:
            cn.append(a[i])
        else:
            cp.append(a[i])
    cp.sort()
    cp.reverse()
    cn.sort()
    x=cp[:5]+cn[:5]
    max=-1000000000000000000000000000
    for i in range(len(x)-4):
        for j in range(i+1,len(x)-3):
            for k in range(j+1,len(x)-2):
                for l in range(k+1,len(x)-1):
                    for m in range(l+1,len(x)):
                        if(x[i]*x[j]*x[k]*x[l]*x[m]>max):
                            max=x[i]*x[j]*x[k]*x[l]*x[m]
    cn.reverse()
    x=cp[:5]+cn[:5]
    for i in range(len(x)-4):
        for j in range(i+1,len(x)-3):
            for k in range(j+1,len(x)-2):
                for l in range(k+1,len(x)-1):
                    for m in range(l+1,len(x)):
                        if(x[i]*x[j]*x[k]*x[l]*x[m]>max):
                            max=x[i]*x[j]*x[k]*x[l]*x[m]
    print(max)
