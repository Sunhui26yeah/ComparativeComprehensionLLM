public class PhysicsPractical {
 
	public static void main(String[] args) {
		Scanner s = new Scanner(System.in);
		int t = s.nextInt();
		while(t-->0){
		    int n = s.nextInt();
	
		    long[] arr = new long[n];
	
		    for(int i=0;i<n;i++){
		        arr[i] = s.nextLong();
		     
		       
		    }

		    Arrays.sort(arr);
		    long ans = 1;
 
            ans = arr[0]*arr[1]*arr[2]*arr[3]*arr[4];
            ans = Math.max(ans, arr[0]*arr[1]*arr[2]*arr[3]*arr[n-1]);
            ans = Math.max(ans, arr[0]*arr[1]*arr[2]*arr[n-2]*arr[n-1]);
            ans = Math.max(ans, arr[0]*arr[1]*arr[n-3]*arr[n-2]*arr[n-1]);
            ans = Math.max(ans, arr[0]*arr[n-4]*arr[n-3]*arr[n-2]*arr[n-1]);
            ans = Math.max(ans, arr[n-5]*arr[n-4]*arr[n-3]*arr[n-2]*arr[n-1]);
            System.out.println(ans);
            
            
		}
    }
 
}
