import java.util.*;
import java.lang.*;
import java.io.*;

/* Name of the class has to be "Main" only if the class is public. */
public class Main
{
	public static void main (String[] args)
	{
		Scanner sc=new Scanner (System.in);
		int t =sc.nextInt();
		outer:
		while(t>0)
		{
		    int n=sc.nextInt();
		    long arr[]=new long[n];
		    int count=0;
		    for(int i=0;i<n;i++)
		    {
		        arr[i]=sc.nextLong();
		        if(arr[i]>=0)
		            count++;
		    }
		    Arrays.sort(arr);
		    long ans=arr[n-1];
		    if(count>0)
		    if(arr[0]*arr[1]>=arr[n-3]*arr[n-2])
		    {
		        ans*=arr[0]*arr[1];
		        ans*=Math.max(arr[2]*arr[3],arr[n-3]*arr[n-2]);
		    }
		    else
		    {
		        ans*=arr[n-2]*arr[n-3];
		        ans*=Math.max(arr[0]*arr[1],arr[n-4]*arr[n-5]);
		    }
		    else
		    {
		       ans=arr[n-1]*arr[n-2]*arr[n-3]*arr[n-4]*arr[n-5]; 
		    }
		    System.out.println(ans);
		    t--;
		}
	}
}
