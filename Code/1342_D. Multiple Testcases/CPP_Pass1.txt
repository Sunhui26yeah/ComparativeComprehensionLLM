#include <bits/stdc++.h>
using namespace std;
long long int check(vector<long long int> a, vector<long long int> c,
                    long long int mid) {
  long long int n = a.size();
  long long int k = c.size();
  vector<vector<long long int>> v;
}
int main() {
  long long int n, k;
  cin >> n >> k;
  vector<long long int> a(n);
  vector<long long int> c(k);
  for (long long int i = 0; i < n; i++) cin >> a[i];
  for (long long int i = 0; i < k; i++) cin >> c[i];
  long long int mn = INT_MIN;
  sort(a.begin(), a.end());
  for (long long int i = 0; i < n; i++) {
    long long int x = (n - i) / c[a[i] - 1];
    if ((n - i) % c[a[i] - 1]) x++;
    mn = max(x, mn);
  }
  vector<long long int> v[mn];
  for (long long int i = 0; i < n; i++) {
    v[i % mn].push_back(a[i]);
  }
  cout << mn << "\n";
  for (long long int i = 0; i < mn; i++) {
    cout << v[i].size() << " ";
    for (long long int j = 0; j < v[i].size(); j++) {
      cout << v[i][j] << " ";
    }
    cout << "\n";
  }
}
