import collections
import time
import os
import sys
import bisect
import heapq
from typing import List


if __name__ == '__main__':
    N, K = map(int, input().split())
    A = [int(x) for x in input().split()]
    C = [0] + [int(x) for x in input().split()]
    
    A.sort()
    
    wca = collections.Counter(A)
    vs = list(sorted(wca.keys()))
    suffsum = [0]
    for v in reversed(vs):
        suffsum.append(suffsum[-1] + wca[v])
    suffsum = suffsum[::-1][:-1]
    
    s = 0
    # c = 0
    greater = [0 for _ in range(K+1)]
    if vs[0] <= K:
        greater[vs[0]] = suffsum[0]
    for i in range(len(vs)-1):
        if vs[i] > K:
            break
        for j in range(vs[i]+1, min(vs[i+1], K)+1):
            greater[j] = suffsum[i+1]
    if vs[-1] <= K:
        greater[vs[-1]] = suffsum[-1]
    
    
    # print(greater[1:])
    # print([greater[i]//C[i] if greater[i] % C[i] == 0 else greater[i]//C[i] + 1 for i in range(1, K+1)])
    m = max([greater[i]//C[i] if greater[i] % C[i] == 0 else greater[i]//C[i] + 1 for i in range(1, K+1)])
    ans = [[] for _ in range(m)]
    for i, v in enumerate(A):
        ans[i % m] .append(v)
    
    print(m)
    for v in ans:
        print('{} {}'.format(len(v), ' '.join(map(str, v))))
