from collections import  defaultdict
from sys import stdin,setrecursionlimit
setrecursionlimit(10**6)
import threading
def dfs(node,g,vis):
	vis[node]=1

	for i in g[node]:
		if vis[i]==0:
			dfs(i,g,vis)
def dfs2(node,g,vis,path):
	path.append(node)
	vis[node]=1
	for i in g[node]:
		if vis[i]==0:
			dfs2(i,g,vis,path)
def main():
	def f(a):
		n=len(a)
		g=defaultdict(list)
		good=None
		for i in range(0,len(a)):
			if a[i]==i+1 and good==None:
				# print(a[i],i,"lodu")
				good=a[i]
			g[i+1].append(a[i])
			g[a[i]].append(i+1)
		# print(g,good)
		vis=[0]*(len(a)+1)
		dfs(good,g,vis)
		cnt=0
		for i in range(1,n+1):
			if i==good-1:
				continue
			if vis[i]==0:
				pah=[]
				dfs2(i,g,vis,pah)
				if pah:
					cnt+=1
				a[i-1]=good
		print(cnt)
		print(*a)
	a=input()
	lst=list(map(int,input().strip().split()))
	f(lst)
threading.stack_size(10 ** 8)
t = threading.Thread(target=main)
t.start()
t.join()
