import java.util.*;
public class horses {
        
        static class comp implements Comparator<pair>
{
    public int compare(pair p1,pair p2)
    {
        if(p1.money<p2.money)
        return 1;
        else if(p1.money>p2.money)
        return -1;
        else return 0;
    }
}

static class pair
{
    int money;
    int factor;
    pair(int money,int factor)
    {
        this.money=money;
        this.factor=factor;
    }
}


    public static void main(String args[]) {
        Scanner sc=new Scanner(System.in);
        int n=sc.nextInt();
		int d=sc.nextInt();
		PriorityQueue<pair> pq=new PriorityQueue<pair>(new comp());
		for(int i=0;i<n;i++)
		pq.add(new pair(sc.nextInt(),sc.nextInt()));
		pair last=pq.poll();
		long prev=last.money;
		long ans=last.factor;
		ArrayList<Long> l=new ArrayList<Long>(); 
		while(pq.size()>0)
		{
			pair temp=pq.poll();
			if(prev-temp.money<d)
			{
				ans+=temp.factor;
			}
			else
			{
				l.add(ans);
				prev=temp.money;
				ans=temp.factor;
			}
		}
		l.add(ans);
		
		Collections.sort(l);
		System.out.println(l.get(l.size()-1));
			
		
      
    }
}
