n,d=map(int,input().split())
l=[]
for i in range(n):
    j=tuple(map(int,input().split()))
    l.append(j)
k=0  
m=0
for i in range(n):
    c=l[i][1]
    for j in range(n):
        if(l[j][0]<l[i][0]+d-1 and l[j][0]+d-1>l[i][0] and i!=j):
            c+=l[j][1]
            #print(l[j][0]+d,l[i][0])
    #print("\n")
    if(c>m):
        m=c
        k=i
print(m)
