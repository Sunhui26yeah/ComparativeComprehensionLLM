class Pair {
	int coin;
	int frdship;

	public Pair() {

	}

	public Pair(int c, int f) {
		coin = c;
		frdship = f;
	}
}

class TaskA {

	public void solve(int testnumber, FastReader in, PrintWriter out) {

		int n = in.ni();
		int d = in.ni();
		
		Pair arr[] = new Pair[n];
		for(int i =0;i<n;i++){
			arr[i] = new Pair(in.ni(), in.ni());
		}
		
		Arrays.sort(arr, new Comparator<Pair>() {

			@Override
			public int compare(Pair o1, Pair o2) {
				return o1.coin- o2.coin;
			}
		});
		
		int start = 0, end = 1;
		
		long max = 0;
		
		long current = arr[0].frdship;
		
		while(start < arr.length){
			
			while(end < arr.length && (arr[start].coin + d) > arr[end].coin){
				current += arr[end].frdship;
				end++;
				
			}
			
			if(current > max){
				max = current;
			}
			current -= arr[start].frdship;
			start++;
		}
		
		out.println(max);
	}
}

class FastReader {
	public InputStream stream;
	private byte[] buf = new byte[1024];
	private int curChar;
	private int numChars;
	private SpaceCharFilter filter;

	public FastReader(InputStream stream) {
		this.stream = stream;
	}

	public FastReader() {

	}

	public int read() {
		if (numChars == -1) {
			throw new InputMismatchException();
		}
		if (curChar >= numChars) {
			curChar = 0;
			try {
				numChars = stream.read(buf);
			} catch (IOException e) {
				throw new InputMismatchException();
			}
			if (numChars <= 0) {
				return -1;
			}
		}
		return buf[curChar++];
	}

	public int ni() {
		int c = read();
		while (isSpaceChar(c))
			c = read();
		int sgn = 1;
		if (c == '-') {
			sgn = -1;
			c = read();
		}
		int res = 0;
		do {
			if (c < '0' || c > '9') {
				throw new InputMismatchException();
			}
			res *= 10;
			res += c - '0';
			c = read();
		} while (!isSpaceChar(c));
		return res * sgn;
	}

	public String ns() {
		int c = read();
		while (isSpaceChar(c))
			c = read();
		StringBuilder res = new StringBuilder();
		do {
			res.appendCodePoint(c);
			c = read();
		} while (!isSpaceChar(c));
		return res.toString();
	}

	public boolean isSpaceChar(int c) {
		if (filter != null) {
			return filter.isSpaceChar(c);
		}
		return isWhitespace(c);
	}

	public static boolean isWhitespace(int c) {
		return c == ' ' || c == '\n' || c == '\r' || c == '\t' || c == -1;
	}

	public int[] iArr(int n) {
		int a[] = new int[n];
		for (int i = 0; i < n; i++) {
			a[i] = ni();
		}
		return a;
	}

	public String next() {
		return ns();
	}

	public interface SpaceCharFilter {
		public boolean isSpaceChar(int ch);
	}

}
