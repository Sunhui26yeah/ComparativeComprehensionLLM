#include <bits/stdc++.h>
using namespace std;
pair<long long int, long long int> ar[100005];
long long int arr[100005];
long long int brr[100005];
long long int cm[100005];
int main() {
  long long int i, n, t, a, b, pos, c, ans, k;
  scanf("%lld%lld", &n, &t);
  for (i = 0; i < n; i++) {
    scanf("%lld%lld", &a, &b);
    ar[i] = make_pair(a, b);
  }
  sort(ar, ar + n);
  for (i = 0; i < n; i++) {
    arr[i] = ar[i].first;
    brr[i] = ar[i].second;
  }
  cm[0] = brr[0];
  for (i = 1; i < n; i++) {
    cm[i] = cm[i - 1] + brr[i];
  }
  arr[n] = -4;
  c = 0;
  ans = 0;
  long long int ll;
  for (i = 0; i < n; i++) {
    ll = t + arr[i] - 1;
    pos = upper_bound(arr, arr + n, ll) - arr;
    pos = pos - 1;
    k = i;
    if (k == 0) {
      c = cm[pos];
    } else {
      c = cm[pos] - cm[k - 1];
    }
    if (c > ans) {
      ans = c;
    }
  }
  printf("%lld", ans);
  return 0;
}
