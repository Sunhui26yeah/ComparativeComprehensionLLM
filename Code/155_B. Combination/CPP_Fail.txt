#include <bits/stdc++.h>
using namespace std;
bool sort_function(const pair<int, int> &a, const pair<int, int> &b) {
  return a.first >= b.first && a.second <= b.second;
}
int main() {
  int n;
  cin >> n;
  vector<pair<int, int> > v;
  for (int i = 0; i < n; i++) {
    int a, b;
    cin >> a >> b;
    v.push_back(make_pair(a, b));
  }
  sort(v.begin(), v.end(), sort_function);
  if (v[n - 1].second == 0) {
    cout << v[0].first << endl;
  } else {
    int score = 0;
    int c1 = 0, c2 = 0;
    int last;
    bool flag = false;
    for (int i = n - 1; i >= 0; i--) {
      if (v[i].second == 0) {
        last = i;
        flag = true;
        c1 += c2;
        break;
      } else {
        score += v[i].first;
        if (c1 == 0) {
          c1 = v[i].second;
        } else if (c1 - 1 == 0) {
          c1 = c2 + v[i].second;
          c2 = 0;
        } else {
          c1--;
          c2 += v[i].second;
        }
      }
    }
    if (!flag) {
      cout << score << endl;
    } else {
      for (int j = 0; j < n && j <= last && c1 > 0; j++, c1--) {
        score += v[j].first;
      }
      cout << score << endl;
    }
  }
}
