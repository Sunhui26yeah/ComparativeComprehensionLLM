#include <bits/stdc++.h>
using namespace std;
int32_t main() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);
  long long n, x, y;
  cin >> n;
  vector<pair<long long, long long>> v;
  for (long long i = 0; i < n; i++) {
    cin >> x >> y;
    v.push_back(make_pair(y, x));
  }
  sort(v.begin(), v.end());
  reverse(v.begin(), v.end());
  long long ans = 0;
  long long cnt = 1;
  for (long long i = 0; i < n; i++) {
    if (cnt == 0) break;
    ans += v[i].second;
    cnt--;
    cnt += v[i].first;
  }
  cout << ans << endl;
  return 0;
}
