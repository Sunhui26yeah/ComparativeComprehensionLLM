import java.util.Arrays;
import java.util.Scanner;

public class RidingInLift {

	public static void main(String[] args) {
		
		int mod=(int) (1e9+7);
		
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		int a=sc.nextInt();
		int b=sc.nextInt();
		int k=sc.nextInt();
		int[][] dp=new int[k+1][n+2];
		for(int i=0;i<=k;i++)
		{
			Arrays.fill(dp[i], 0);
		}
		dp[0][a]=1;
		for(int i=1;i<=k;i++)
		{
			for(int j=1;j<=n;j++)
			{
				if(dp[i-1][j]==0)
					continue;
				int abs=Math.abs(j-b)-1;
				int maxfloor=Math.min(n, j+abs);
				int minfloor=Math.max(1, j-abs);
				dp[i][minfloor]=(dp[i][minfloor]+dp[i-1][j]+mod)%mod;
				dp[i][j]=(dp[i][j]-dp[i-1][j]+mod)%mod;
				dp[i][j+1]=(dp[i][j+1]+dp[i-1][j]+mod)%mod;
				dp[i][maxfloor+1]=(dp[i][maxfloor]-dp[i-1][j]+mod)%mod;
			}
			for(int j=1;j<=n;j++)
			{
				dp[i][j]=dp[i][j]+dp[i-1][j];
			}
		}
		int answer=0;
		for(int i=1;i<=n;i++)
			answer+=dp[k][i];
		System.out.println(answer%mod);
	}
	
}
