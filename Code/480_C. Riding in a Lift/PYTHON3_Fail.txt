n, a, b, k = [int(n) for n in input().split()] 

a = a - 1
b = b - 1

#A = [ min(max(i, n-i)-1, max(abs(b - i), 0)) for i in range(n) ]

A = [ max(abs(b - i) - 1, 0) for i in range(n) ] # dist from lab
B = [ min(i,n-i - 1) for i in range(n) ] # dist from borders
C = [ A[i] + min(A[i], B[i]) for i in range(n) ]

#print(A)
#print(B)
#print(C)

R = C

for i in range(1,k):
	for j in range(n):
		for l in range(1, abs(b - j)):
			if A[j + l]:
				R[j] = A[j]*A[j+l] % 1000000007

print(R[a])
