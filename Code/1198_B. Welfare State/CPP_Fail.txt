#define debx cout<<" X "<<endl;
#define deby cout<<" Y "<<endl;
#define debz cout<<" Z "<<endl;
using namespace std;

struct query{
    public:
    ll p,x,y;
};

int main(){
    ll n,q;
    cin>>n;
    ll i,a[n];
    for(i=0;i<n;i++)
        cin>>a[i];
    bool v[n+5]={0};
    cin>>q;
    query b[q+5];
    for(i=0;i<q;i++){
        ll p;
        cin>>p;
        if(p==1){
            ll x,y;
            cin>>x>>y;
            b[i].p=p;
            b[i].x=x-1;
            b[i].y=y;
            a[x-1]=y;
        }
        else{
            ll x;
            cin>>x;
            b[i].p=p;
            b[i].x=x;
        }
    }
    ll currlow=-1,maxx=0;
    for(i=q-1;i>=0;i--){
        if(b[i].p==1){
            ll x,y;
            x=b[i].x;
            y=b[i].y;
            if(v[x])
                continue;
            v[x]=1;
            if(y<currlow)
                y=currlow;
            a[x]=y;
        }
        else{
            currlow=b[i].x;
            maxx=max(maxx,currlow);
        }
    }
    for(i=0;i<n;i++){
        if(v[i]==0){
            a[i]=max(maxx,a[i]);
        }
        cout<<a[i]<<" ";
    }
}
