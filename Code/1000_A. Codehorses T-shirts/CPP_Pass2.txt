int cpmlen(const string &a, const string &b) {
  if (a.size() == b.size()) return a < b;
  return a.size() < b.size();
}
int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.precision(10);
  int n;
  cin >> n;
  vector<string> a(n), b(n);
  for (auto &ai : a) cin >> ai;
  for (auto &bi : b) cin >> bi;
  sort((a).begin(), (a).end(), cpmlen);
  sort((b).begin(), (b).end(), cpmlen);
  map<char, int> m;
  for (int i = 0; i < (n); ++i)
    if (a[i].size() == 1) m[a[i][0]]++, m[b[i][0]]--;
  int ans = abs(m['M']) + abs(m['L']) + abs(m['S']);
  ans /= 2;
  for (int i = 0; i < (n); ++i)
    if (a[i].size() > 1) ans += a[i] != b[i];
  cout << ans;
  return 0;
}
