import java.util.*;
public class A1000 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String[] s1 = new String[n];
        String[] s2 = new String[n];
        for (int i = 0; i < n; i++) {
            s1[i] = sc.next();
        }
        String s;
        Boolean b;
        int x = 0;
        for (int i = 0; i < n; i++) {
            s = sc.next();
            b = true;
            for (int j = 0; j < n; j++) {
                if(s.equals(s1[j])){
                    b = false;
                    s1[j] = "ZZZZZ";
                    x++;
                    s2[i] = "ZZZZZ";
                    break;
                }
            }
            if(b){
                s2[i] = s;
            }
        }
        Arrays.sort(s1);
        Arrays.sort(s2);
        int ans = 0;

        for (int i = 0; i < n - x; i++) {
            if(s1[i].charAt(s1[i].length() - 1) != s2[i].charAt(s2[i].length() - 1)){
                ans++;
            }
        }
        System.out.println(ans);
    }
    
}
