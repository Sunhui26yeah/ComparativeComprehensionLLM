                    a[i] = '1'
                #print(i)
        else:
            if b[i]=='0':
                if (a[i-1]=='1' and b[i-1] == '1') or (a[i-1]=='0' and b[i-1]=='0'):
                    a[i] = '1'
            else:
                if (a[i-1]=='1' and b[i-1] == '0') or (a[i-1]=='0' and b[i-1]=='1') or (a[i-1]=='0' and b[i-1]=='0'):
                    a[i] = '1'
