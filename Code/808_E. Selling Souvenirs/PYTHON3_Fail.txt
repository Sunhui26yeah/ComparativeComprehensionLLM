def knapsack(items, weight):
    n = len(items)
    dp = [[0] * (weight+1) for _ in range(n)]
    for r in range(0, n):
        cur_w, cur_cost = items[r]
        for w in range(1, weight+1):
            best = 0
            if r > 0:
                best = dp[r-1][w]
            if cur_w == w:
                best = max(best, cur_cost)
            if w - cur_w >= 0:
                best = max(best, dp[r-1][w-cur_w] + cur_cost)
            best = max(best, dp[r][w-1])
            dp[r][w] = best
    return dp[-1][-1]

n, m = map(int, input().split())
items = []
for _ in range(n):
    items.append(tuple(map(int, input().split())))

items.sort(key=lambda t: (t[1] / t[0], -t[0]))
cost = 0
while m > 120 and n:
    w, c = items.pop()
    cost += c
    m -= w
    n -= 1

if not items:
    print(cost)
    sys.exit()

print(cost + knapsack(items, m))
