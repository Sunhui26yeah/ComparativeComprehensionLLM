import java.io.*;
import java.util.*;

public class LastCorollaryLooseLoop implements Runnable {
    public static void main(String[] args) throws Exception {
        new Thread(null, new LastCorollaryLooseLoop(), ": )", 1 << 28).start();
    }

    public void run() {
        FastScanner sc = new FastScanner();
        PrintWriter out = new PrintWriter(System.out);

        N = sc.nextInt();
        int M = sc.nextInt();
        int k = sc.nextInt();
        adjList = new ArrayList[N];
        ArrayDeque<Integer> edges = new ArrayDeque<>();
        Arrays.setAll(adjList, i -> new ArrayList<>());
        for (int i = 0; i < M; i++) {
            int u = sc.nextInt() - 1;
            int v = sc.nextInt() - 1;
            edges.add(u);
            edges.add(v);
            adjList[u].add(v);
            adjList[v].add(u);
        }
        groups = new ArrayDeque[2];
        Arrays.setAll(groups, i -> new ArrayDeque<>());
        cycle = new ArrayDeque<>();
        inCycle = new boolean[N];
        parent = new int[N];
        depth = new int[N];
        vis = new int[N];
        dfs(0, -1, 0, 0);
//        System.out.println(Arrays.toString(parent));
//        System.out.println(Arrays.toString(depth));
        int need = (k + 1) / 2;
        if (cycle.isEmpty()) {
            // then it's a tree!
            int which = groups[0].size() > groups[1].size() ? 0 : 1;
            StringBuilder ans = new StringBuilder();
            while (need > 0) {
                ans.append(groups[which].pollFirst() + 1).append(" ");
                need--;
            }
            out.println(1);
            out.println(ans);
        } else {
            // need to shave down the cycle
            while (edges.size() > 0) {
                int u = edges.pollFirst();
                int v = edges.pollFirst();
                // the edge can't just be a direct link though
                if (inCycle[u] && inCycle[v] && Math.abs(depth[u] - depth[v]) >= 2) {
                    // burn off both ends until they meet this edge.
                    // because the ends of the cycle are one apart, they
                    // can NOT span this edge, so they will not
                    // degenerate into a single node. So, they must end up
                    // at different point, which will preserve the cycle's
                    // nondegeneracy
                    while (!(cycle.getFirst() == u || cycle.getFirst() == v)) {
                        // keep burning it down
                        inCycle[cycle.pollFirst()] = false;
                    }
                    while (!(cycle.getLast() == u || cycle.getLast() == v)) {
                        inCycle[cycle.pollLast()] = false;
                    }
                }
            }
            // now, we found a tight loop. same as previous solution from here
            if (cycle.size() <= k) {
                StringBuilder ans = new StringBuilder();
                for (int i : cycle) {
                    ans.append(i + 1).append(" ");
                }
                out.println(2);
                out.println(cycle.size());
                out.println(ans);
            } else {
                // since it's a minimal cycle, we can alternate validly.
                StringBuilder ans = new StringBuilder();
                while (need > 0) {
                    ans.append(cycle.pollFirst() + 1).append(" ");
                    cycle.pollFirst();
                    need--;
                }
                out.println(1);
                out.println(ans);
            }
        }
        out.close();
    }

    static ArrayList<Integer>[] adjList;
    static ArrayDeque<Integer>[] groups;
    static int[] parent, depth;
    static ArrayDeque<Integer> cycle;
    static boolean[] inCycle;
    static int[] vis;
    static int N;

    static void dfs(int node, int par, int color, int d) {
        parent[node] = par;
        groups[color].add(node);
        if (par != -1) depth[node] = depth[par] + 1;
        int back = -1;
        for (int adj : adjList[node]) {
            if (adj != par) {
                if (vis[adj] == 0) {
                    vis[adj] = 1;
                    dfs(adj, node, 1 ^ color, d + 1);
                    vis[adj] = 2;
                } else if (vis[adj] == 1 && depth[adj] <= depth[node] - 2) {
                    back = adj;
                }
            }
        }
        if (cycle.isEmpty() && back != -1) {
            // let's just grab any back edge, then later we'll trim it down!
            int curr = node;
            while (curr != back) {
                cycle.add(curr);
                inCycle[curr] = true;
                curr = parent[curr];
            }
            cycle.add(curr); // last bit
            inCycle[curr] = true;
        }
    }

    static class FastScanner {
        private int BS = 1 << 16;
        private char NC = (char) 0;
        private byte[] buf = new byte[BS];
        private int bId = 0, size = 0;
        private char c = NC;
        private double cnt = 1;
        private BufferedInputStream in;

        public FastScanner() {
            in = new BufferedInputStream(System.in, BS);
        }

        public FastScanner(String s) {
            try {
                in = new BufferedInputStream(new FileInputStream(new File(s)), BS);
            } catch (Exception e) {
                in = new BufferedInputStream(System.in, BS);
            }
        }

        private char getChar() {
            while (bId == size) {
                try {
                    size = in.read(buf);
                } catch (Exception e) {
                    return NC;
                }
                if (size == -1) return NC;
                bId = 0;
            }
            return (char) buf[bId++];
        }

        public int nextInt() {
            return (int) nextLong();
        }

        public int[] nextInts(int N) {
            int[] res = new int[N];
            for (int i = 0; i < N; i++) {
                res[i] = (int) nextLong();
            }
            return res;
        }

        public long[] nextLongs(int N) {
            long[] res = new long[N];
            for (int i = 0; i < N; i++) {
                res[i] = nextLong();
            }
            return res;
        }

        public long nextLong() {
            cnt = 1;
            boolean neg = false;
            if (c == NC) c = getChar();
            for (; (c < '0' || c > '9'); c = getChar()) {
                if (c == '-') neg = true;
            }
            long res = 0;
            for (; c >= '0' && c <= '9'; c = getChar()) {
                res = (res << 3) + (res << 1) + c - '0';
                cnt *= 10;
            }
            return neg ? -res : res;
        }

        public double nextDouble() {
            double cur = nextLong();
            return c != '.' ? cur : cur + nextLong() / cnt;
        }

        public double[] nextDoubles(int N) {
            double[] res = new double[N];
            for (int i = 0; i < N; i++) {
                res[i] = nextDouble();
            }
            return res;
        }

        public String next() {
            StringBuilder res = new StringBuilder();
            while (c <= 32) c = getChar();
            while (c > 32) {
                res.append(c);
                c = getChar();
            }
            return res.toString();
        }

        public String nextLine() {
            StringBuilder res = new StringBuilder();
            while (c <= 32) c = getChar();
            while (c != '\n') {
                res.append(c);
                c = getChar();
            }
            return res.toString();
        }

        public boolean hasNext() {
            if (c > 32) return true;
            while (true) {
                c = getChar();
                if (c == NC) return false;
                else if (c > 32) return true;
            }
        }
    }

    static void ASSERT(boolean assertion, String message) {
        if (!assertion) throw new AssertionError(message);
    }

    static void ASSERT(boolean assertion) {
        if (!assertion) throw new AssertionError();
    }
}
