import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;


public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader r = new BufferedReader(new InputStreamReader(System.in));
        
        String[] line = r.readLine().split("[ ]+");
        int N = Integer.parseInt(line[0]);
        int K = Integer.parseInt(line[1]);
        
        char[][] a = new char[2][];
        a[0] = r.readLine().toCharArray();
        a[1] = r.readLine().toCharArray();
        
        boolean[][] v = new boolean[N + K + 10][2];
        
        Queue<State> q = new LinkedList<State>();
        q.add(new State(0, 0, 0, 0));
        
        while(!q.isEmpty()){
            State st = q.poll();
//          System.out.println(st.i + ", " + st.d + ", " + st.w);
            
            if(st.i >= N){
                System.out.println("YES");
                return;
            }
                
            if(st.i + 1 >= N || (a[st.d][st.i + 1] == '-' && !v[st.i + 1][st.d])){
                v[st.i + 1][st.d] = true;
                if(st.w + 1 <= st.i + 1)q.add(new State(st.i + 1, st.d, 1 - st.t, st.w + 1));
            }
            if(st.i - 1 >= 0 && (a[st.d][st.i - 1] == '-' && !v[st.i - 1][st.d])){
                v[st.i - 1][st.d] = true;
                if(st.w + 1 <= st.i - 1)q.add(new State(st.i - 1, st.d, 1 - st.t, st.w + 1));
            }
            if(st.i + K >= N || (a[1 - st.d][st.i + K] == '-' && !v[st.i + K][1 - st.d])){
                v[st.i + K][1 - st.d] = true;
                if(st.w + 1 <= st.i + K)q.add(new State(st.i + K, 1 - st.d, 1 - st.t, st.w + 1));
            }
        }
        
        System.out.println("NO");
    }
    
}
class State{
    int i, d, t, w;
    public State(int ii, int di, int ti, int wi){
        i = ii;
        d = di;
        t = ti;
        w = wi;
    }
}
