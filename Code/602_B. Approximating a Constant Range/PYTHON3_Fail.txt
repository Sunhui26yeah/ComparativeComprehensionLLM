def main():
    input()
    seq = [int(c) for c in input().split()]

    last_diff = last_idx = start = ans = 0
    for i in range(1, len(seq)):
        diff = seq[i] - seq[i-1]
        if diff:
            if diff == last_diff and last_diff:
                start = last_idx

            last_idx = i

        ans = max(ans, i - start + 1)

    print(ans)


if __name__ == "__main__":
    main()
