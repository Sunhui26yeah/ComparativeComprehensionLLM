#include <bits/stdc++.h>
using namespace std;
int a[100010];
int main() {
  int t;
  scanf("%d", &t);
  while (t--) {
    long long n, k;
    scanf("%lld %lld", &n, &k);
    string s;
    cin >> s;
    for (int i = 1; i <= n; i++) a[i] = s[i - 1] - '0';
    for (int i = 1; i <= n; i++) {
      if (a[i] == 0) {
        int j = i;
        while (j) {
          if (k == 0) break;
          if (j - 1 != 0) swap(a[j], a[j - 1]), k--;
          j--;
        }
      }
      if (k == 0) break;
    }
    for (int i = 1; i <= n; i++) printf("%d", a[i]);
    printf("\n");
  }
}
