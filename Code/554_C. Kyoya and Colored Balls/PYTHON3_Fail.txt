k=int(input())
l=[int(input()) for i in range(k)]
s=0 
ans=[0]*k 
ans[0]=1 
MOD=10**9+7 
fact=[0]*(10**6+5)
fact[0]=1 
for i in range(1,10**6+5):
    fact[i]=(fact[i-1]%MOD*i%MOD)%MOD
def c(n,k):
    if k>=n:
        return 1 
    if k==0 or k==n:
        return 1 
    return (fact[n]//(fact[k]*fact[n-k]))%MOD 
ans=1 
sm=l[0]
for i in range(1,k):
    curr=l[i]
    ans=ans*(c(sm+curr-1,curr-1))
    ans%=MOD 
    sm+=l[i]
print(ans)
