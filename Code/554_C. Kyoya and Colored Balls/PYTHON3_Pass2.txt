def fact(x):
    res = 1
    while x > 0:
        res *= x
        x -= 1
    return res

k = int(input())
lst=[]
for i in range(0, k):
    x = int(input())
    lst.append(x)

l = lst[0]
ans = 1
for i in range(1, k):
    l += lst[i] - 1
    ans *= fact(l) // (fact(l - lst[i] + 1) * fact(lst[i] - 1))
    l += 1
print(ans % (1000 * 1000 * 1000 + 7))
