#include <bits/stdc++.h>
using namespace std;
int a[300][300], b[300][300];
int dir[4][2] = {{0, 1}, {1, 0}, {-1, 0}, {0, -1}};
int main() {
  int n, t;
  while (scanf("%d %d", &n, &t) != EOF) {
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    a[150][150] = n;
    int tag = 1;
    while (tag) {
      tag = 0;
      memset(b, 0, sizeof(b));
      for (int i = 0; i < 300; i++) {
        for (int j = 0; j < 300; j++) {
          if (a[i][j] >= 4) {
            tag = 1;
            for (int k = 0; k < 4; k++) {
              b[i + dir[k][0]][j + dir[k][1]] = a[i][j] / 4;
            }
            b[i][j] += a[i][j] % 4;
          }
        }
      }
      if (tag) memcpy(a, b, sizeof(b));
    }
    while (t--) {
      int x, y;
      scanf("%d %d", &x, &y);
      x += 150;
      y += 150;
      if (x < 300 && y < 300 && x >= 0 && y >= 0) {
        printf("%d\n", a[x][y]);
      } else {
        puts("0");
      }
    }
  }
  return 0;
}
