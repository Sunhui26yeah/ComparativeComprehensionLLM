from sys import *
f = lambda: map(int, stdin.readline().split())
n, t = f()

m = 65
r = range(1, m)
p = [[0] * m for i in range(m)]
p[1][0] = n // 4
p[0][0] = n % 4

s = k = 1
while s:
    s = 0

    for x in r[:k]:
        if p[x][0] > 3:
            s = 1
            d = p[x][0] // 4
            p[x][0] %= 4

            p[x + 1][0] += d
            p[x][1] += d

            if x != 1:
                p[x - 1][0] += d
            else:
                p[1][1] += d
                p[1][0] += d
        for y in r[:x - 1]:
            if p[x][y] > 3:
                s = 1
                d = p[x][y] // 4
                p[x][y] %= 4

                p[x + 1][y] += d
                p[x - 1][y] += d
                p[x][y + 1] += d
                p[x][y - 1] += d

                if x == y + 1:
                    p[x][x] += d
                    p[y][y] += d
                if y == 1: p[x][0] += d
        if p[x][x] > 3:
            s = 1
            d = p[x][x] // 4
            p[x][x] %= 4

            p[x + 1][x] += d
            p[x][x - 1] += d

            if x == 1: p[1][0] += d
    k += 1

s = []
for j in range(t):
    x, y = f()
    x, y = abs(x), abs(y)
    if x < y: x, y = y, x
    s.append(p[x][y] if x < m else 0)

stdout.write('\n'.join(map(str, s)))
