import java.util.*;

public class CodeForces598C {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        int m = scanner.nextInt();
        int d = scanner.nextInt();
        int[] c = new int[m];
        int rest = n;
        for (int i = 0; i < m; i++) {
            c[i] = scanner.nextInt();
            rest -= c[i];
        }
        if (rest > (d - 1) * (m + 1)) {
            System.out.println("NO");
        } else {
            StringBuilder sb = new StringBuilder("YES\n");
            boolean river = rest > 0;
            int riverStep = 0;
            int j = 0;
            for (int i = 0; i < n;) {
                if (river) {
                    sb.append("0 ");
                    riverStep++;
                    rest--;
                    if (riverStep == d - 1 || rest == 0) {
                        river = false;
                    }
                    i++;
                } else {
                    for (int k = 0; k < c[j]; k++) {
                        sb.append(j + 1).append(" ");
                        i++;
                    }
                    j++;
                    riverStep = 0;
                    if (rest > 0) river = true;
                }
            }
            System.out.println(sb);
        }
    }
}
