#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e6 + 2;
int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  cout.tie(NULL);
  long long int q;
  cin >> q;
  while (q--) {
    long long int i, j, n, m, r, ans = 0;
    cin >> n >> r;
    map<long long int, bool> found;
    long long int arr[n + 1];
    for (i = 0; i < n; ++i) cin >> arr[i];
    sort(arr, arr + n);
    long long int impt = 0;
    for (i = n - 1; i >= 0; --i) {
      if (found[arr[i]]) continue;
      arr[i] += impt;
      if (arr[i] >= 0 && !found[arr[i]]) {
        ++ans;
        impt -= r;
        found[arr[i]] = true;
      }
    }
    cout << ans << "\n";
  }
}
