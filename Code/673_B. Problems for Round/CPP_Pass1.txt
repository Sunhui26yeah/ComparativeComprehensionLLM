#include <bits/stdc++.h>
using namespace std;
int manj[111111];
int vec[111111];
int main() {
  int n, m, ans = 0;
  scanf("%d%d", &n, &m);
  if (m == 0) {
    printf("%d\n", n - 1);
    return 0;
  }
  int maks1 = 0, min2 = n;
  for (int i = 0; i < m; i++) {
    int a, b;
    scanf("%d%d", &a, &b);
    if (a > b) swap(a, b);
    manj[a]++;
    vec[b]++;
    maks1 = max(a, maks1);
    min2 = min(b, min2);
  }
  for (int i = 1; i <= n; i++)
    if (manj[i] >= 1 && vec[i] >= 1) {
      printf("0");
      return 0;
    }
  ans = min2 - maks1;
  if (min2 <= maks1) {
    printf("0");
    return 0;
  } else
    printf("%d", ans);
}
