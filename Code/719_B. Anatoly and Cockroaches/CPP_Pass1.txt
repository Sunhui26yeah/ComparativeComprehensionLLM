#include <bits/stdc++.h>
using namespace std;
string in, rb, br;
int n, iler, ileb, wynikrb, wynikbr, zle, pozb, pozr;
int main() {
  ios_base::sync_with_stdio(0);
  cin >> n;
  cin >> in;
  for (int i = 0; i < n; i++) {
    if (in[i] == 'r') {
      iler++;
    } else {
      ileb++;
    }
  }
  for (int i = 0; i < n; i++) {
    if (in[i] == 'r' && i % 2 == 1) {
      zle++;
      pozr++;
    }
    if (in[i] == 'b' && i % 2 == 0) {
      zle++;
      pozb++;
    }
  }
  wynikrb += max(pozb, pozr);
  zle = 0;
  pozr = 0;
  pozb = 0;
  for (int i = 0; i < n; i++) {
    if (in[i] == 'r' && i % 2 == 0) {
      zle++;
      pozr++;
    }
    if (in[i] == 'b' && i % 2 == 1) {
      zle++;
      pozb++;
    }
  }
  wynikbr += max(pozr, pozb);
  cout << min(wynikrb, wynikbr);
}
