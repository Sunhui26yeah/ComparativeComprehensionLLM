#include <bits/stdc++.h>
using namespace std;
const int N = 10000 + 5;
int vis[N];
int main() {
  memset(vis, 0, sizeof(vis));
  int n, m, flag = 0;
  cin >> m >> n;
  for (int i = 0; i < n; i++) {
    int a, k, flag0 = 1;
    cin >> k;
    for (int j = 0; j < k; j++) {
      cin >> a;
      if (vis[abs(a)]) flag0 = 0;
      vis[abs(a)] = 1;
    }
    if (flag0) flag = 1;
    memset(vis, 0, sizeof(vis));
  }
  if (flag)
    printf("YES\n");
  else
    printf("NO\n");
  return 0;
}
