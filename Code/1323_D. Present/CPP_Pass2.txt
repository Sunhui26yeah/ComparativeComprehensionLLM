#include <bits/stdc++.h>
using namespace std;
const int maxn = 4e5 + 50;
int t, n, m;
int arr[maxn];
int main() {
  scanf("%d", &n);
  for (int i = 1; i <= n; i++) {
    scanf("%d", &arr[i]);
  }
  long long fin = 0;
  for (int k = 16777216; k > 0; k >>= 1) {
    for (int i = 1, w = k * 2 - 1; i <= n; i++) {
      arr[i] = arr[i] & w;
    }
    sort(arr + 1, arr + n + 1);
    int cnt[3] = {1, -1, 1};
    int rec[3] = {n + 1, n + 1, n + 1};
    int now = 0;
    for (int i = 1; i <= n; i++) {
      for (int j = 0; j < 3; j++) {
        for (int v = (j + 1) * k; rec[j] > 1 && arr[i] + arr[rec[j] - 1] >= v;
             --rec[j])
          ;
      }
      int sum = 0;
      for (int j = 0; j < 3; j++) sum += cnt[j] * max(0, i - rec[j]);
      now ^= (sum & 1);
    }
    fin += now * k;
  }
  printf("%lld\n", fin);
  return 0;
}
