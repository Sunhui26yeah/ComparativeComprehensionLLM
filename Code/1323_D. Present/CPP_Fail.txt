#include <bits/stdc++.h>
using namespace std;
const int N = 4e5 + 5;
const int M = 5e7 + 5;
int a[N];
long long cnt[M];
int ans = 0;
int main() {
  int n;
  scanf("%d", &n);
  long long sum = 0;
  for (int i = 1; i <= n; i++) {
    scanf("%d", &a[i]);
    if (a[i] & 1) {
      sum++;
    }
  }
  if ((sum * (n - sum)) % 2 == 1) {
    ans += 1;
  }
  for (int i = 2; i <= 24; i++) {
    int l = (1 << (i - 2)), mid = (1 << (i - 1)), r = 1 << i;
    for (int j = l - 1; j < r; j++) {
      cnt[j] = 0;
    }
    for (int j = 1; j <= n; j++) {
      cnt[a[j] & (r - 1)]++;
    }
    long long sum = 0;
    for (int j = l; j < r; j++) {
      int mn = mid - j - 1, mx = min(j - 1, r - 1 - j);
      if (mn < 0) {
        sum += cnt[j] * cnt[mx];
      } else if (mx >= mn) {
        sum += cnt[j] * (cnt[mx] - cnt[mn]);
      }
      cnt[j] += cnt[j - 1];
    }
    if (sum & 1) {
      ans += mid;
    }
  }
  memset(cnt, 0, sizeof(cnt));
  for (int i = 1; i <= n; i++) {
    cnt[a[i]]++;
  }
  for (int i = 1; i <= n; i++) {
    if (cnt[a[i]]) {
      if (((cnt[a[i]] * (cnt[a[i]] - 1)) / 2) & 1) {
        ans ^= a[i] + a[i];
      }
      cnt[a[i]] = 0;
    }
  }
  printf("%d", ans);
}
