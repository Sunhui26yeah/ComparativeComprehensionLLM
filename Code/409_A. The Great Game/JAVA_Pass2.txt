import java.util.Scanner;

public class TheGreatGame {
    public static void main(String[] args) {
        Scanner cin = new Scanner(System.in);
        int score = 0;
        char[] t1 = cin.next().toCharArray();
        char[] t2 = cin.next().toCharArray();

        for (int i = 0; i < t2.length; i += 2) {
            if (t1[i] == t2[i])
                continue;

            if (t1[i] == '[') {
                if (t2[i] == '(') {
                    score += 1;
                } else {
                    score += -1;
                }
            } else if (t1[i] == '(') {
                if (t2[i] == '8') {
                    score += 1;
                } else {
                    score += -1;
                }
            } else {
                if (t2[i] == '[') {
                    score += 1;
                } else {
                    score += -1;
                }
            }
        }
        if (score > 0) {
            System.out.println("TEAM 1 WINS");
        } else if (score < 0) {
            System.out.println("TEAM 2 WINS");
        } else {
            System.out.println("TIE");
        }
        cin.close();
    }
}
