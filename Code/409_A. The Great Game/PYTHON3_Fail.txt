s1=input()
s2=input()

if(len(s1)%2==0):
    x=len(s1)//2
    a1=s1[x-1]+s1[x]
    a2=s2[x-1]+s2[x]
    if(a1>a2):
        print("TEAM 1 WINS")
    elif(a2>a1):
        print("TEAM 2 WINS")
    else:
        print("TIE")
else:
    x=len(s1)//2
    a1=s1[x-1]+s1[x]+s1[x+1]
    a2=s2[x-1]+s2[x]+s2[x+1]
    if(a1>a2):
        print("TEAM 1 WINS")
    elif(a2>a1):
        print("TEAM 2 WINS")
    else:
        print("TIE")
