import sys


def solve():
    ok = "OK"
    wrong = "WRONG_ANSWER"
    n = input()
    m = input()
    if len(n) != len(m): return wrong
    if len(n) == 1:
        if n == m: return ok
        return wrong
    res = list()
    smallestfirst = 10
    for c in n:
        if int(c) < smallestfirst and int(c) > 0: smallestfirst = int(c)
    if smallestfirst == 10: return wrong
    res.append(smallestfirst)
    for c in n:
        if int(c) == 0:
            res.append(0)
    n = list(n)
    rem = list()
    n.remove(str(smallestfirst))
    for c in n:
        if int(c) != 0:
            rem.append(int(c))
    rem.sort()
    for i in rem:
        res.append(i)
    correct = ''.join(map(str,res))
    if correct == m: return ok
    return wrong


def read(mode=2):
    inputs = input().strip()
    if mode == 0: return inputs  # String
    if mode == 1: return inputs.split()  # List of strings
    if mode == 2: return list(map(int, inputs.split()))  # List of integers


def write(s="\n"):
    if s is None: s = ""
    if isinstance(s, list): s = " ".join(map(str, s))
    if isinstance(s, tuple): s = " ".join(map(str, s))
    s = str(s)
    print(s, end="")


def run():
    if sys.hexversion == 50594544: sys.stdin = open("test.txt")
    res = solve()
    write(res)

run()
