  sort(A.begin(), A.end());
  int a = 0, first = 1;
  for (int i = 0; i < A.size(); i++)
    if (A[i] != '0') {
      a = a * 10 + (A[i] - '0');
      if (first) {
        for (int j = 0; j < A.size(); j++)
          if (A[j] == '0') a = a * 10;
        first = 0;
      }
    }
  cout << a << endl;
