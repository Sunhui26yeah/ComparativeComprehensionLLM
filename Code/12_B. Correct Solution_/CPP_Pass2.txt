#include <bits/stdc++.h>
using namespace std;
char sl = '\n';
const int nsize = 101;
int number[10];
int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  cout.tie(nullptr);
  string n, m;
  cin >> n >> m;
  for (char i : n) number[i - '0']++;
  int mi = 0;
  for (int i = 1; i < 10; ++i) {
    if (number[i]) {
      mi *= 10;
      mi += i;
      number[i]--;
      break;
    }
  }
  for (int i = 0; i < number[0]; ++i) mi *= 10;
  for (int i = 1; i < 10; ++i) {
    for (int j = 0; j < number[i]; ++j) {
      mi *= 10;
      mi += i;
    }
  }
  int b = 0;
  for (int i = 0; i < m.size(); ++i) {
    b *= 10;
    b += (m[i] - '0');
  }
  cout << (mi == b && m.size() == n.size() ? "OK" : "WRONG_ANSWER") << sl;
  return 0;
}
