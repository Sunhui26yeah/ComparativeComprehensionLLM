#include <bits/stdc++.h>
using namespace std;
int n, k;
int a[100005];
int f[100005];
int h[100005];
int play[100005];
int card[100005];
long long dp[505][5005];
long long rec(int i, int j) {
  if (i == 0 || j == 0) return 0;
  if (dp[i][j] != -1) return dp[i][j];
  long long s = 0;
  for (int q = 0; q <= k; q++)
    if (q <= j) s = max(s, rec(i - 1, j - q) + h[q]);
  return dp[i][j] = s;
}
int main() {
  scanf("%d %d", &n, &k);
  for (int i = 0; i < n * k; i++) scanf("%d", a + i);
  for (int i = 0; i < n; i++) scanf("%d", f + i);
  for (int i = 1; i <= k; i++) scanf("%d", h + i);
  for (int i = 0; i < n; i++) play[f[i]]++;
  for (int i = 0; i < n * k; i++) card[a[i]]++;
  long long ans = 0;
  for (int i = 0; i < 100005; i++)
    if (play[i] && card[i]) {
      memset(dp, -1, sizeof dp);
      ans += rec(play[i], card[i]);
    }
  printf("%I64d\n", ans);
}
