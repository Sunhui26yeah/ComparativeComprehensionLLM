#include <bits/stdc++.h>
using namespace std;
int a[100010], s[100010];
int main() {
  int i, j, k, n, ans = 0;
  scanf("%d", &n);
  for (i = 1; i <= n; i++) scanf("%d", &a[i]);
  s[n] = a[n];
  for (i = n - 1; i >= 1; i--) s[i] = min(a[i], s[i + 1]);
  for (i = 1; i < n; i++)
    if (a[i] > s[i + 1]) ans++;
  printf("%d\n", ans);
  return 0;
}
