t = int(input())
 
while t:
    t -= 1
    n, k  = map (int, input().split())
    balls = []
    for i in range (n):
        x, y = map(int, input().split())
        balls.append([x,y])
        
    mx = my = float('inf')
    Mx = My = float ('-inf')
    for i in range (n):
        mx = min(balls[i][0], mx)
        my = min (balls[i][1], my)
        Mx = max (balls[i][0], Mx)
        My = max (balls[i][1], My)
    d = abs(Mx-mx)+abs(My-my)
    if d>k:
        print (-1)
        continue
    else:
        print (1)
        continue
