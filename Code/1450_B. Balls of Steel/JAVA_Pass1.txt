import java.util.Scanner;

public class BallsOfSteel {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int t = in.nextInt();
        while (t-- > 0) {
            int n, k, i, j, count = 0;
            n = in.nextInt();
            k = in.nextInt();
            int[][] a = new int[n][2];
            //int[] b=new int[n];
            for (i = 0; i < n; i++) {                
                    a[i][0]=in.nextInt();
                    a[i][1]=in.nextInt();                
            }

            for (i = 0; i < n; i++) {
                count = 0;
                for (j = 0; j < n; j++) {
                    if (i != j && Math.abs(a[i][0] - a[j][0]) + Math.abs(a[i][1] - a[j][1]) <= k) {
                        count++;
                    }
                }
                if (count == n - 1) {
                    System.out.println(1);
                    break;
                }
            }
            if (i == n) {
                System.out.println("-1");
            }
        }
    }
}
