#include <bits/stdc++.h>
using namespace std;
void solve() {
  long long int n, k, i, j, x, y;
  cin >> n >> k;
  long long int a[n][2];
  double b[n][n - 1];
  for (i = 0; i < n; i++) {
    for (j = 0; j < 2; j++) cin >> a[i][j];
  }
  for (i = 0; i < n; i++) {
    long long int l = -1;
    for (j = 0; j < n; j++) {
      if (i == j) continue;
      x = a[i][0] - a[j][0];
      y = a[i][1] - a[j][1];
      b[i][++l] = sqrt((x * x) + (y * y));
    }
  }
  for (i = 0; i < n; i++) {
    for (j = 0; j < n - 1; j++) {
      if (b[i][j] > k) break;
    }
    if (j == n - 1) {
      cout << "1" << endl;
      return;
    }
  }
  cout << "-1" << endl;
}
int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  long long int t;
  cin >> t;
  while (t--) solve();
}
