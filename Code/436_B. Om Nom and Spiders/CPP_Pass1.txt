#include <bits/stdc++.h>
using namespace std;
int main() {
  int n, m, k;
  cin >> n >> m >> k;
  vector<string> a(n);
  for (int i = (0); i < (n); ++i) {
    cin >> a[i];
  }
  vector<int> res(m, 0);
  for (int i = (0); i < (n); ++i) {
    for (int j = (0); j < (m); ++j) {
      if (a[i][j] == '.') continue;
      if (a[i][j] == 'U') {
        if (i % 2 == 0) res[j]++;
      }
      if (a[i][j] == 'D') continue;
      if (a[i][j] == 'L') {
        if (j - i >= 0) res[j - i]++;
      }
      if (a[i][j] == 'R') {
        if (j + i < m) res[j + i]++;
      }
    }
  }
  for (int i = (0); i < (m); ++i) cout << res[i] << " ";
  return 0;
}
