def f():
    n, m, k = map(int, input().split())
    a = []
    for i in range(n):
        a.append(list(input().strip()))
    res = [0 for i in range(n)]
    for i in range(m):
        for j in range(1, n):
            if j < i and a[j][i - j] == 'R':
                res[i] += 1
            if j < n - i and a[j][i + j] == 'L':
                res[i] += 1
            if j + j < m and i < n and a[j + j][i] == 'U':
                res[i] += 1
    print(' '.join(list(map(str, res))))
f()
