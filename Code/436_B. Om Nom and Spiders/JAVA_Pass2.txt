	public static void main(String args[]) {
		Scanner scan = new Scanner(System.in);
		
		String cpt[] = scan.nextLine().split(" ");
		int n = Integer.parseInt(cpt[0]);
		int m = Integer.parseInt(cpt[1]);
		int k = Integer.parseInt(cpt[2]);
		
		ArrayList<Integer> counts = new ArrayList<Integer>(m);
		for (int x = 0; x < m; x++) {
			counts.add(0);
		}

		for (int i = 0; i < n; i++) {
			String l = scan.nextLine();
			for (int j = 0; j < m; j++) {
				char c = l.charAt(j);
				if (c == 'U') {
					if (i % 2 == 0) {
						counts.set(j, counts.get(j) + 1);
					}
				} else if (c == 'R') {
					if (j + i < m) {
						counts.set(i + j, counts.get(i + j) + 1);
					}
				} else if (c == 'L') {
					if (j - i >= 0) {
						counts.set(j - i, counts.get(j - i) + 1);
					}
				}
			}
		}

		StringBuilder sb = new StringBuilder();
		for (int count : counts) {
			sb.append(count + " ");
		}
		
		System.out.println(sb);
	}
