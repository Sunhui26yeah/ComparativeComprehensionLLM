bool cmp(pair<ll,ll> a,pair<ll,ll> b){
    if(a.first != b.first){
        return a.first < b.first;
    }else{
        return a.second < b.second;
    }
}

int main()
{  
    #ifndef ONLINE_JUDGE

    #endif
    IOS;
    ll t;
    cin >> t;
    while(t--){
        ll n; 
        cin >> n;
        vector<string> s(n);
        for(ll i = 0;i < n;i++){
                cin >> s[i];
        } 
        ll k= 0;
        for(ll i = 0;i < n;i++){
            for(ll j = 0;j < n;j++){
                if (s[i][j] != '.'){ 
                    k++;
                }
            }
        }
        bool flag = true;
        for(ll x = 0;x < 3;x++){
            for(ll y = 0;y < 3;y++){
                if(x == y){
                    continue;
                }else{
                    ll count = 0;
                    for(ll i = 0;i < n;i++){
                        for(ll j= 0;j < n;j++){
                            if(s[i][j] == 'X' && (i + j)%3 == x){
                                count++;
                            }else if(s[i][j] == 'O' && (i + j)%3 == y){
                                count++;
                            }
                        }
                    }
                    if(count <= k/3){
                        for(ll i = 0;i < n;i++){
                            for(ll j= 0;j < n;j++){
                                if(s[i][j] == 'X' && (i + j)%3 == x){
                                    s[i][j] = 'O';
                                }else if(s[i][j] == 'O' && (i + j)%3 == y){
                                    s[i][j] = 'X';
                                }
                            }
                        }
                        for(auto it : s){
                            cout << it << "\n";
                        }
                        flag = false;
                        break;
                    }

                }
            }
            if(!flag){
                break;
            }
        }
    }
    return 0;
}
