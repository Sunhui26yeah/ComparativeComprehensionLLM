#include <bits/stdc++.h>
using namespace std;
int main() {
  ios::sync_with_stdio(false);
  int n, m;
  cin >> n >> m;
  set<int> st1[51], st2, st3;
  vector<string> vec(n + 1);
  for (int i = 0; i < n; i++) {
    cin >> vec[i + 1];
  }
  int a, b, c;
  for (int i = 1; i <= n; i++) {
    st2.clear();
    for (int j = 0; j < m; j++) {
      if (vec[i][j] == '#') {
        st2.insert(j + 1);
      }
    }
    for (int k = 1; k <= n; k++) {
      if (k != i) {
        if (st2 != st1[k]) {
          st3 = st2;
          while (st3.size()) {
            a = *st3.begin();
            if (st1[k].find(a) != st1[k].end()) {
              cout << "No";
              return 0;
            } else {
              st3.erase(a);
            }
          }
        } else {
          break;
        }
      }
    }
    st1[i] = st2;
  }
  cout << "Yes";
  return 0;
}
