#include <bits/stdc++.h>
#pragma gcc optimize(o3)
using namespace std;
const long long N = 2e5 + 5;
map<long long, bool> mop;
bool isPrime(long long x) {
  for (long long i = 2; i * i <= x; i++) {
    if (x % i == 0) return 0;
  }
  return 1;
}
void getDivs(long long x) {
  for (long long i = 2; i * i <= x; i++) {
    if (x % i == 0 and isPrime(i)) mop[i] = mop[x / i] = 1;
  }
  mop[x] = 1;
}
int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);
  long long a, b, n;
  cin >> n >> a >> b;
  getDivs(a);
  getDivs(b);
  for (int i = 1; i < n; i++) {
    cin >> a >> b;
    for (auto &i : mop) {
      if (i.second and a % i.first and b % i.first) i.second = 0;
    }
  }
  for (auto &i : mop) {
    if (i.second) return cout << i.first, 0;
  }
  cout << -1;
  return 0;
}
