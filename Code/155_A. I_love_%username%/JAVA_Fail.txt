import java.util.Scanner;

public class Task155A {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int[] scores = new int[n];
		for (int i = 0; i < n; i++) {
			scores[i] = sc.nextInt();
		}
		sc.close();
		
		int max = scores[1];
		int min = scores[1];
		int miracle = 0;
		for (int i = 0; i < n; i++) {
			if (max < scores[i]) {
				max = scores[i];
				miracle++;
			}
			if (min < scores[i]) {
				min = scores[i];
				miracle++;
			}
		}
		
		System.out.println(miracle);
	}
}
