#include <bits/stdc++.h>
using namespace std;
int main() {
  int N;
  cin >> N;
  int arr[101][101];
  vector<int> vec;
  int tmp;
  for (int i = 0; i < N; i++)
    for (int j = 0; j < N; j++)
      if (i == j) {
        cin >> arr[i][j];
        arr[i][j] = 0;
      } else
        cin >> arr[i][j];
  if (N > 1) {
    for (int i = 0; i < N; i++) {
      tmp = 0;
      for (int j = 0; j < N; j++) {
        tmp ^= arr[i][j];
      }
      cout << tmp << " ";
    }
  } else
    cout << "0";
  cout << endl;
}
