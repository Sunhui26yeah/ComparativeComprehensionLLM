#include <bits/stdc++.h>
using namespace std;
int n, b[105][105], a[105];
int main() {
  cin >> n;
  for (int i = 0; i < n; ++i)
    for (int j = 0; j < n; ++j) {
      cin >> b[i][j];
      if (b[i][j] != -1) {
        a[i] |= b[i][j];
        a[j] |= b[i][j];
      }
    }
  for (int i = 0; i < n; ++i) cout << max(a[i], 0) << ' ';
  cout << endl;
}
