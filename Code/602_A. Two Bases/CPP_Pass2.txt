#include <bits/stdc++.h>
using namespace std;
long long v(vector<long long> x, long long b) {
  long long p = 1, ans = 0;
  for (int i = x.size() - 1; i >= 0; i--) {
    ans += x[i] * p;
    p *= b;
  }
  return ans;
}
int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);
  ;
  long long n, m, bx, by, vx, vy;
  cin >> n >> bx;
  vector<long long> x(n);
  for (int i = 0; i < n; i++) cin >> x[i];
  cin >> m >> by;
  vector<long long> y(m);
  for (int i = 0; i < m; i++) cin >> y[i];
  vx = v(x, bx);
  vy = v(y, by);
  if (vx > vy) {
    cout << '>' << endl;
    return 0;
  }
  if (vx == vy) {
    cout << '=' << endl;
    return 0;
  }
  cout << '<' << endl;
  return 0;
}
