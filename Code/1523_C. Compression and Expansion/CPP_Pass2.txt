ll gcd(ll a ,ll b)
{
    
    while(b!=0)
    {
       ll rem = a%b;
        a=b;
        b=rem;
        
    }
    return a;
}
ll moduloexp(ll x, ll y , ll m) // o(logy)  returns x^y%m
{
    
    if(y==0)
    {
        return 1;
    }
    if(y%2==0)
    {
        return (moduloexp(x,y/2,m)*(moduloexp(x,y/2,m)))%m;
    }
    else
    {
        return((x%m)*(moduloexp(x,y-1,m)))%m;
    }
    
}
ll modulofacto(ll n , ll p) //returns (n!)%p
{
    if (n>=p)
    {
        return 0;
    }
    else{
        ll result =1;
        for(ll i=1;i<=n;i++)
        {
            result = (result*i)%p;
        }
        return result;
    }
}
