	public static void solve(FastReader fs) {
		int n = fs.nextInt();
		int arr[] = new int[n];
		for (int i = 0; i < n; i++) {
			arr[i] = fs.nextInt();
		}
		StringBuilder out = new StringBuilder();
		ArrayList<Integer> stack = new ArrayList<>();
		for (int i = 0; i < n; i++) {
			if (arr[i] == 1)
				stack.add(arr[i]);
			else {
				if (stack.size() > 0) {
					int x = arr[i] - 1;
					for (int u = stack.size() - 1; u >= 0; u--) {
						if (stack.get(u) == x) {
							stack.set(u, arr[i]);
							break;
						} else {
							stack.remove(u);
						}
					}
				}
			}
			int j = 0;
			for (int val : stack) {
				out.append(val);
				if (j != stack.size() - 1) {
					out.append(".");
				}
				j++;
			}
			out.append("\n");
		}
		System.out.print(out);
	}




	static long power(long a, long b) {
		long result = 1;
		if (b == 0)
			return result;
		while (b > 0) {
			if ((b & 1) == 1)result *= a;
			a *= a;
			b /= 2;
		}
		return result;
	}


//IO operation
	static class FastReader {
		BufferedReader br;
		StringTokenizer st;
		public FastReader() {
			br = new BufferedReader(
			    new InputStreamReader(System.in));
		}

		String next() {
			while (st == null || !st.hasMoreElements()) {
				try {
					st = new StringTokenizer(br.readLine());
				} catch (IOException e) {
					e.printStackTrace();
				}
			}
			return st.nextToken();
		}

		int nextInt() { return Integer.parseInt(next()); }

		long nextLong() { return Long.parseLong(next()); }

		double nextDouble() {
			return Double.parseDouble(next());
		}

		String nextLine() {
			String str = "";
			try {
				str = br.readLine();
			} catch (IOException e) {
				e.printStackTrace();
			}
			return str;
		}
	}
}
