from typing import List

def solve(n: int, a: List[int]) -> None:
    stack = []
    for i in range(n):
        num = a[i]
        if (not stack):
            stack.append(num)
        else:
            if (num == 1):
                stack.append(num)
            elif (num <= stack[-1]):
                while (stack and num - stack[-1] != 1):
                    stack.pop()
                if (not stack):
                    stack.append(num)
                else:
                    stack[-1] = num
            else:
                while (stack and num - stack[-1] != 1):
                    stack.pop()
                if (not stack):
                    stack.append(num)
                else:
                    stack[-1] = num

        l = ""
        for x in stack:
            l += str(x) + "."
        l = l[:-1]
        print(l)
    return
