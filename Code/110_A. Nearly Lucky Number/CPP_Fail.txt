#include <bits/stdc++.h>
int main() {
  std::ios::sync_with_stdio(false);
  long long int n;
  std::cin >> n;
  bool isLucky = true;
  while (n != 0) {
    if (n % 10 != 4 && n % 10 != 7) {
      isLucky = false;
      break;
    }
    n = n / 10;
  }
  if (isLucky)
    std::cout << "YES";
  else
    std::cout << "NO";
}
