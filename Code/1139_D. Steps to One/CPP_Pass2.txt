#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 10;
const int mod = 1e9 + 7;
int n, cnt, pri[N], mu[N];
bool bz[N];
long long ans, fac[N], ifac[N], inv[N];
long long power(long long x, int n) {
  long long a = 1;
  while (n) {
    if (n & 1) a = a * x % mod;
    x = x * x % mod;
    n >>= 1;
  }
  return a;
}
void init(int n) {
  mu[1] = 1;
  for (int i = 2; i <= n; ++i) {
    if (!bz[i]) pri[++cnt] = i, mu[i] = mod - 1;
    for (int j = 1; j <= cnt && pri[j] * i <= n; ++j) {
      bz[i * pri[j]] = 1;
      if (!(i % pri[j])) {
        mu[i * pri[j]] = 0;
        break;
      }
      mu[i * pri[j]] = (-mu[i] + mod) % mod;
    }
  }
  fac[0] = 1;
  for (int i = 1; i <= n; ++i) fac[i] = fac[i - 1] * i % mod;
  ifac[n] = power(fac[n], mod - 2);
  for (int i = n; i >= 1; --i) ifac[i - 1] = ifac[i] * i % mod;
  for (int i = 1; i <= n; ++i) inv[i] = ifac[i] * fac[i - 1] % mod;
}
int dec(int a, int b) { return a >= b ? a - b : a - b + mod; }
int main() {
  scanf("%d", &n);
  init(n);
  ans = 1;
  for (int i = 2; i <= n + 1; ++i) {
    ans = dec(ans, inv[n - n / i] * (n / i) % mod * mu[i] % mod);
  }
  printf("%lld\n", ans);
  return 0;
}
