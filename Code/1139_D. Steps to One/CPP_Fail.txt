#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9 + 7;
const int N = 1e5 + 9;
long long power(long long a, long long b, long long m = mod) {
  long long res = 1;
  while (b) {
    if (b & 1) {
      res = 1LL * res * a % m;
    }
    a = 1LL * a * a % m;
    b /= 2;
  }
  return res;
}
vector<int> pf[N];
int f_cnt(int n, int a) {
  int sz = pf[a].size();
  int ret = 0;
  for (int msk = 0; msk < (1 << sz); msk++) {
    int tmp = 1;
    int cnt = 0;
    for (int bit = 0; bit < sz; bit++) {
      if ((msk >> bit) & 1) {
        tmp *= pf[a][bit];
        cnt++;
      }
    }
    if (cnt & 1)
      ret = ret - n / tmp;
    else
      ret = ret + n / tmp;
  }
  return ret;
}
vector<int> divs[N];
int lp[N];
long long f[N];
signed main() {
  ios_base::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);
  vector<int> pr;
  for (int i = 2; i < N; ++i) {
    if (lp[i] == 0) {
      lp[i] = i;
      pr.push_back(i);
    }
    for (int j = 0; j < (int)pr.size() && pr[j] <= lp[i] && i * pr[j] < N; ++j)
      lp[i * pr[j]] = pr[j];
  }
  int m;
  cin >> m;
  for (int i = 2; i <= m; i++) {
    int u = i;
    while (u > 1) {
      int k = lp[u];
      while (u % k == 0) u /= k;
      pf[i].push_back(k);
    }
  }
  for (int i = 1; i < N; i++) {
    for (int j = 2 * i; j < N; j += i) {
      divs[j].push_back(i);
    }
  }
  f[1] = 1;
  for (int i = 2; i <= m; i++) {
    f[i] = m;
    for (int j : divs[i]) {
      f[i] += f[j] * f_cnt(m / j, i / j);
    }
    f[i] %= mod;
    f[i] = f[i] * power((m - m / i + mod) % mod, mod - 2, mod) % mod;
  }
  long long ans = 0;
  for (int i = 1; i <= m; i++) {
    ans += f[i];
  }
  cout << ans * power(m, mod - 2, mod) % mod;
  return 0;
}
