n = int(input())
arr = [*map(int, input().split())]

s1 = set([*range(n + 1)])
s2 = set(arr)
s1 = s1 - s2

s1 = [*s1][::-1]
s2 = [*s2][::-1]

ans = [s1[-1]]

del s1[-1]

for i in range(1, n):
    if arr[i] != arr[i - 1]:
        ans.append(s2[-1])
        del s2[-1]
    else:
        ans.append(s1[-1])
        del s1[-1]

print(*ans)
