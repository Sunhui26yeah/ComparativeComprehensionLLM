import sys
input = sys.stdin.readline
N = int(input())
a = list(map(int, input().split()))

res = [-1] * N
if a[0]: res[0] = 0
for i in range(N - 1):
  if a[i + 1] > a[i]: res[i + 1] = a[i]

s = sorted(set([x for x in range(N + 1)]) - set(a) - set(res), reverse = True)
for i in range(N):
  if res[i] == -1: res[i] = s.pop()

print(*res)
