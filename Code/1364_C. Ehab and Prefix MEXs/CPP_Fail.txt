const int N = 1e6 + 10, M = 1e3 + 10, mod = 1e9 + 7;
const double eps = 1e-6, Pi = acos(-1.0);
int a[N], b[N], vis[N], mark[N];
set<int> st;
int main() {
  int n;
  in(n);
  for (int i = 1; i <= n; i++) {
    in(a[i]);
    vis[a[i]]++;
  }
  for (int i = 0; i < N; i++) {
    if (!vis[i]) {
      st.insert(i);
    }
  }
  for (int i = 1; i <= n; i++) {
    auto it = st.upper_bound(a[i]);
    if (it != st.begin()) {
      it--;
    }
    b[i] = *it;
    st.erase(it);
    if (vis[a[i]] == 1) {
      st.insert(a[i]);
    }
    vis[a[i]]--;
  }
  int p = 0, v = 0;
  for (int i = 1; i <= n; i++) {
    mark[b[i]] = 1;
    while (mark[p]) p++;
    if (p != a[i]) v = 1;
    p++;
  }
  if (v) return 0 * printf("-1\n");
  for (int i = 1; i <= n; i++) {
    printf("%d ", b[i]);
  }
  return 0;
}
