#include <bits/stdc++.h>
using namespace std;
const int N = 201000;
const double eps = 1e-8;
const int inf = 1e9 + 7;
const long long mod = 1e9 + 7;
long long p[N];
long long gcd(long long a, long long b) {
  if (!b) return a;
  return gcd(b, a % b);
}
void solve() {
  long long i, a, b, q, l, r, ans, g;
  cin >> a >> b >> q;
  if (a > b) swap(a, b);
  g = a / gcd(a, b) * b;
  for (i = 0; i < g; i++) p[i + 1] = p[i] + (i % a % b != i % b % a);
  while (q--) {
    cin >> l >> r;
    ans = (r - l + 1) / g * p[g];
    l %= g;
    r %= g;
    if (l <= r)
      ans += p[r + 1] - p[l];
    else
      ans += p[g] - p[l] + p[r + 1];
    cout << ans << ' ';
  }
  cout << '\n';
}
int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  int i, n;
  cin >> n;
  while (n--) solve();
  return 0;
}
