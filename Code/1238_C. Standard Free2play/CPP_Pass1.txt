#include <bits/stdc++.h>
using namespace std;
int main(void) {
  int q, x[200001];
  cin >> q;
  while (q-- > 0) {
    int h, n, ans = 0;
    cin >> h >> n;
    for (int i = 0; i < n; i++) {
      cin >> x[i];
    }
    x[n] = 0;
    for (int i = 0; i < n; i++) {
      if (x[i + 1] <= 1) break;
      if (x[i + 2] == x[i + 1] - 1) {
        i++;
      } else {
        ans++;
        x[i + 1]--;
      }
    }
    cout << ans << endl;
  }
}
