import java.util.*;
import java.math.*;

public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int t=sc.nextInt();
		StringBuilder anso=new StringBuilder("");
		while(t-->0)
		{
		    int h=sc.nextInt();
		    int n=sc.nextInt();
		    int[] arr=new int[n];
		    for(int i=0;i<n;i++)
		    arr[i]=sc.nextInt();
		    int[] ans=new int[n];
		    Arrays.fill(ans,Integer.MAX_VALUE);
		    ans[n-1]=0;
		    for(int i=n-2;i>=0;i--)
		    {
		        int x=arr[i+1]+1;
		        if(i==(n-2))
		        {
		            if(x==2)
		            ans[i]=0;
		            else
		            ans[i]=1;
		        }
		        else
		        {
		           if(arr[i+1]==(arr[i+2]+1))
		           {
		               ans[i]=Math.min(ans[i+1]+1,ans[i+2]);
		           }
		           else
		           {
		               ans[i]=ans[i+1]+1;
		           }
		        }
		    }
		    anso.append(ans[0]+"\n");
		}
		System.out.println(anso);
	}
}
