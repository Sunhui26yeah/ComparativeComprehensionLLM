#include <bits/stdc++.h>
using namespace std;
const int N = 2e5 + 100;
const int mod = 1e9 + 7;
void hey() {
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);
}
int main() {
  hey();
  int q;
  cin >> q;
  while (q--) {
    int h, n;
    cin >> h >> n;
    int a[n + 1];
    map<int, int> was;
    for (int i = 1; i <= n; i++) {
      cin >> a[i];
      was[a[i]] = 1;
    }
    int ans = 0;
    for (int i = 2; i <= n; i++) {
      if (h <= 2) {
        break;
      }
      if (h - a[i] == 1) {
        if (was[a[i] - 1]) {
          h = a[i] - 1;
        } else {
          ans++;
          h = a[i] - 1;
        }
      } else {
        h = a[i] + 1;
        i--;
      }
    }
    cout << ans << '\n';
  }
}
