#include<bits/stdc++.h>
using namespace std;
int main(){
int a,b;
cin>>a>>b;
char c[a][b];
int d[a+2][b+2];
for(int i=0;i<a;i++){
for(int j=0;j<b;j++){
cin>>c[i][j];
if(c[i][j]=='.') d[i][j]=0;
if(c[i][j]=='#') d[i][j]=1;}
}
for(int i=0;i<a;i++){
for(int j=0;j<b;j++){
if(d[i][j]==1&&(i==0||d[i-1][j]==0)&&(i==a-1||d[i+1][j]==0)&&(j==0||d[i][j-1]==0)&&(j==b-1||d[i][j+1]==0)){
cout<<"No";
return 0;}
}
}
cout<<"Yes";
return 0;
}
