template <class T>
void pv(T a, T b) {
  for (T i = a; i != b; ++i) cout << (*i) << " ";
  cout << endl;
}
template <class T>
void chmin(T& a, const T& b) {
  if (a > b) a = b;
}
template <class T>
void chmax(T& a, const T& b) {
  if (a < b) a = b;
}
int nextInt() {
  int x;
  scanf("%d", &x);
  return x;
}
long long nextLong() {
  long long x;
  scanf("%lld", &x);
  return x;
}
int main2() {
  int H = nextInt();
  int W = nextInt();
  vector<string> s(H);
  for (int i = 0, _n = (int)(H); i < _n; ++i) cin >> s[i];
  bool yes = true;
  for (int i = 0, _n = (int)(H); i < _n; ++i)
    for (int j = 0, _n = (int)(W); j < _n; ++j) {
      if (s[i][j] == '#') {
        int k = 0;
        if (j < W && s[i][j + 1] == '#') k++;
        if (j - 1 >= 0 && s[i][j - 1] == '#') k++;
        if (i < H && s[i + 1][j] == '#') k++;
        if (i - 1 >= 0 && s[i - 1][j] == '#') k++;
        if (k == 0) yes = false;
      }
    }
  cout << (yes ? "YES" : "NO") << endl;
  return 0;
}
int main() {
  main2();
  return 0;
