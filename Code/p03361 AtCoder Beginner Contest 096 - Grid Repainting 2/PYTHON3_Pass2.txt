h,w=map(int,input().split())
l = [input() for i in range(h)]
a = "."*(w+2)
ans="Yes"
for _ in range(h):
  l[_]="."+l[_]+"."
l.append(a)
l.insert(0,a)
for i in range(1,h+1):
  for j in range(1,w+1):
    if l[i][j]=="#" and l[i-1][j]=="." and l[i][j-1]=="." and l[i+1][j]=="." and l[i][j+1]==".":
      ans="No"
print(ans)
