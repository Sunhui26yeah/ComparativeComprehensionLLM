#include <bits/stdc++.h>
using namespace std;
int32_t main() {
  int n, I;
  cin >> n >> I;
  int t = 1;
  vector<int> A(n);
  for (int i = 0; i < n; i++) {
    cin >> A[i];
  }
  unordered_set<int> s;
  unordered_map<int, int> um;
  for (int i = 0; i < n; i++) {
    s.insert(A[i]);
    um[A[i]]++;
  }
  int min1 = INT_MAX;
  for (int i = 0; i < n; i++) {
    min1 = min(min1, um[A[i]]);
  }
  int len = s.size();
  int len1 = len;
  long long ans1 = 1, ans2 = 1;
  int an = n;
  int K = I * 8;
  int y = K / n;
  ans1 = y > 30 ? 1 << 30 : 1 << y;
  int k = 0;
  while (len >= 2) {
    k++;
    len = len / 2;
  }
  if (len <= ans1) {
    cout << 0;
  } else {
    k = K / n;
    while (k) {
      t = t * 2;
      k--;
    }
    cout << (len1 - t) * min1;
  }
  return 0;
}
