import sys
import math
input = sys.stdin.readline
n,I=map(int,input().split())
a=list(map(int,input().split()))
k=int(8*I//n)
K = 2**k
a.sort()
b=[]
b.append(0)
for i in range(1,n):
    if a[i]!=a[i-1]:
        b.append(i)
#print(b)
if len(b) <= K:
    print(0)
else:
    print(n - max(b[i + K] - b[i] for i in range(len(b) - K)))
