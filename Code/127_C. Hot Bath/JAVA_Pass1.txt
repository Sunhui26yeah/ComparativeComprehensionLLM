import java.util.*;
import java.io.*;
public class C127 {
    public static long gcd(long a, long b) {
        return (b == 0) ? a : gcd(b, a%b);
    }
    public static class Rat {
        long n, d;
        Rat(long n, long d) {
            long gcd = gcd(n,d);
            this.n=n/gcd; this.d=d/gcd;
        }
        public Rat sub(long t0) {
            return new Rat(n - t0*d, d);
        }
        public int compareTo(Rat r) {
            long diff = n*r.d - r.n*d;
            if (diff < 0) return -1;
            if (diff > 0) return 1;
            return 0;
        }
    }
    public static Rat ZERO = new Rat(0, 1);
    public static void main(String[] args) throws IOException {
        Scanner s = new Scanner(System.in);
        long t1 = s.nextInt(), t2 = s.nextInt(), x1 = s.nextInt(), x2 = s.nextInt(), t0 = s.nextInt();
        Rat min = new Rat(1,0);
        long miny1 = 0, miny2 = 0;
        for (long y1 = 0; y1 <= x1; y1++) {
            long num = (t0-t1)*y1;
            long den = (t2-t0);
            long ceil;
            if (den == 0) ceil = x2;
            else ceil = Math.min(x2, (num+den-1)/den);
            if (y1 == 0 && ceil == 0) ceil = x2;
            Rat r = new Rat(t1*y1+t2*ceil,y1+ceil).sub(t0);
            if (r.compareTo(ZERO)<0) continue;
            if (r.compareTo(min)<0) { min = r; miny1 = y1; miny2 = ceil; }
            else if (r.compareTo(min)==0 && (miny1+miny2 < y1+ceil)) { miny1 = y1; miny2 = ceil; }
        }
        System.out.println(miny1+" "+miny2);
    }
}
