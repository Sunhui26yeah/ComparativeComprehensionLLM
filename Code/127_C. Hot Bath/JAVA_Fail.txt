import java.io.BufferedReader;
import java.io.FileReader;
import java.io.InputStreamReader;
import java.io.PrintStream;
import java.io.PrintWriter;
import java.util.StringTokenizer;

public class ProblemC {

    /**
     * @param args
     */
    public static void main(String[] args) throws Exception {
        new ProblemC();
    }

    public ProblemC() throws Exception {
        BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
        PrintStream out = System.out;
        StringTokenizer tok = new StringTokenizer(in.readLine());
        int t1 = Integer.parseInt(tok.nextToken());
        int t2 = Integer.parseInt(tok.nextToken());
        int x1 = Integer.parseInt(tok.nextToken());
        int x2 = Integer.parseInt(tok.nextToken());
        int t0 = Integer.parseInt(tok.nextToken());
        int y1 = Integer.MAX_VALUE;
        int y2 = Integer.MAX_VALUE;
        if (t0 == t2) {
            y1 = 0;
            y2 = x2;
        } else if (t0 == t1) {
            y1 = x1;
            y2 = 0;
        } else {
            for (int y = x1; y >= 1; y--) {
                int x = 0;
                double a = (1.0 * y * (t1 - t0) / (t0 - t2));
                if (t0 > t2) {
                    x = (int) a;
                } else {
                    x = (int) Math.ceil(a);
                }
                double b = 1.0 * (t1 * y + t2 * x) / (y + x);
                double b2 = 1.0 * (t1 * y1 + t2 * y2) / (y1 + y2);
                if (b >= t0 && (b - t0 < b2 - t0 || y1 == Integer.MAX_VALUE)) {
                    y1 = y;
                    y2 = x;
                }
            }
        }
        out.println(y1 + " " + y2);
    }
}
