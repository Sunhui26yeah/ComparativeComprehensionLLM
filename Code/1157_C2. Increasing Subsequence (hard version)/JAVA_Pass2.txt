import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class C {

	static int n;
	static int[] a;
	
	public static void main(String[] args) {
		FastReader scan = new FastReader();
		n = scan.nextInt();
		a = new int[n];
		int max = 0;
		for(int i = 0; i < n; i++) {
			a[i] = scan.nextInt();
		}
		int i = 0, j = n - 1;
		StringBuilder res = new StringBuilder("");
		while(i <= j) {
			if(a[i] > max && a[j] > max) {
				if(a[i] < a[j]) {
					res.append("L");
					max = a[i];
					i++;
				}
				else if(a[j] < a[i]){
					res.append("R");
					max = a[j];
					j--;
				}
				else {
					int aa = checkA(i, j, a);
					int bb = checkB(i, j, a);
					if(aa > bb) {
						for(int l = 0; l <= aa; l++) res.append("L");
					}
					else {
						for(int l = 0; l <= bb; l++) res.append("R"); 
					}
					break;
				}
			}
			else if(a[i] > max) {
				res.append("L");
				max = a[i];
				i++;
			}
			else if(a[j] > max) {
				res.append("R");
				max = a[j];
				j--;
			}
			else break;
		}
		System.out.println(res.length());
		System.out.println(res);
	}
	
	static int checkA(int i, int j, int[] a) {
		int count = 0;
		int max = a[i];
		for(int k = i; k < j; k++) {
			if(a[k] < a[k+1]) count++;
			else break;
		}
		return count;
	}
	static int checkB(int i, int j, int[] a) {
		int count = 0;
		int max = a[i];
		for(int k = j; k > i; k--) {
			if(a[k] < a[k-1]) count++;
			else break;
		}
		return count;
	}
	
	static class Path {
		int len;
		StringBuilder s;
		
		public Path(String a, int b) {	
			s.append(a);
			len = b;
		}
	}
	
	static class FastReader 
    { 
        BufferedReader br; 
        StringTokenizer st; 
  
        public FastReader() 
        { 
            br = new BufferedReader(new
                     InputStreamReader(System.in)); 
        } 
  
        String next() 
        { 
            while (st == null || !st.hasMoreElements()) 
            { 
                try
                { 
                    st = new StringTokenizer(br.readLine()); 
                } 
                catch (IOException  e) 
                { 
                    e.printStackTrace(); 
                } 
            } 
            return st.nextToken(); 
        } 
  
        int nextInt() 
        { 
            return Integer.parseInt(next()); 
        } 
  
        long nextLong() 
        { 
            return Long.parseLong(next()); 
        } 
  
        double nextDouble() 
        { 
            return Double.parseDouble(next()); 
        } 
  
        String nextLine() 
        { 
            String str = ""; 
            try
            { 
                str = br.readLine(); 
            } 
            catch (IOException e) 
            { 
                e.printStackTrace(); 
            } 
            return str; 
        } 
    }

}
