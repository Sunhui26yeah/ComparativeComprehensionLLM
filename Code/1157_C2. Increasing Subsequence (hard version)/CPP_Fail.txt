#include <bits/stdc++.h>
using namespace std;
const int MOD = (int)1e9 + 7;
const int MAX = 1e6;
int arr[1000000];
int main() {
  string ans;
  int st, en, i, last, a;
  cin >> a;
  for (i = 0; i < a; i++) cin >> arr[i];
  st = 0, en = a - 1, last = 0;
  while (st <= en) {
    if (arr[st] > last && arr[en] > last) {
      if (arr[st] < arr[en]) {
        last = arr[st];
        st++;
        ans += 'L';
      } else {
        last = arr[en];
        en--;
        ans += 'R';
      }
    } else if (arr[st] > last) {
      last = arr[st];
      st++;
      ans += 'L';
    } else if (arr[en] > last) {
      last = arr[en];
      en--;
      ans += 'R';
    } else
      break;
  }
  cout << ans.length() << endl;
  cout << ans << endl;
  return 0;
}
