#include <bits/stdc++.h>
using namespace std;
int a[200005];
string s = "";
void func(int p, int q, int prev) {
  int pr = p, ql = q, t = prev;
  while (pr < q && a[pr] > t) {
    t = a[pr];
    pr++;
  }
  t = prev;
  while (ql > p && a[ql] > t) {
    t = a[ql];
    ql--;
  }
  int d1 = pr - p;
  int i;
  int d2 = q - ql;
  if (d1 >= d2) {
    for (i = 1; i <= d1; i++) s += "L";
  } else {
    for (i = 1; i <= d2; i++) s += "R";
  }
}
int main() {
  int n, p, q, i, prev = 0;
  cin >> n;
  for (i = 1; i <= n; i++) cin >> a[i];
  p = 1;
  q = n;
  while (p < q) {
    if (a[p] < a[q]) {
      if (a[p] > prev) {
        s += "L";
        prev = a[p];
        p++;
      } else if (a[q] > prev) {
        s += "R";
        prev = a[q];
        q--;
      } else
        break;
    } else if (a[q] < a[p]) {
      if (a[q] > prev) {
        s += "R";
        prev = a[q];
        q--;
      } else if (a[p] > prev) {
        s += "L";
        prev = a[p];
        p++;
      } else
        break;
    } else {
      func(p, q, prev);
      break;
    }
  }
  if (p == q && a[p] > prev) s += "L";
  cout << s.length() << endl;
  cout << s << endl;
}
