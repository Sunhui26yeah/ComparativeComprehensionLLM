import sys
input = sys.stdin.readline
n = int(input())
a = list(map(int,input().split()))
ind1 = a.index(max(a))
ind2 = a.index(min(a))
k = 0
f = False
for i in range(n - 1):
    if a[i] > a[i + 1]:
        k += 1
        if k == 2:
            f = True
            break
if f or ind1 != 0 and ind2 != n - 1:
    print('NO')
else:
    print('YES')
