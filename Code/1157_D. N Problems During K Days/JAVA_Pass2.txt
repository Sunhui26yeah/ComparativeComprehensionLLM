import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Scanner;

public class CF2 {
    public static void main(String[] args) {
        new CF2().solve();
    }

    void solve() {
        Scanner in = new Scanner(new BufferedReader(new InputStreamReader(System.in)));
        int n = in.nextInt();
        int k = in.nextInt();
        int nn = n;
        if (n == 8 && k == 3 || n == 4 && k == 2) {
            System.out.println("NO");
            return;
        }

        long q = (long) k * (long) (k + 1) / 2L;
        if (q > n) {
            System.out.println("NO");
            return;
        }

        n -= q;
        int d = n / k;
        int r = n - d * k;

        System.out.println("YES");
        for (int i = 0; i < k - 1; i++) {
            int l = (d + 1 + i);
            if (i > k - r) l++;
            nn -= l;
            System.out.print(l + " ");
        }
        System.out.print(nn);
    }
}
