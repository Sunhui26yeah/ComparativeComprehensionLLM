#include <bits/stdc++.h>
const int N = 1e5 + 7;
using namespace std;
long long ans[N];
int main() {
  long long i, j, k, n, m, sum, a0;
  scanf("%lld%lld", &n, &k);
  sum = 0;
  for (i = 1; i <= k; i++) {
    sum = sum * 2 + 1;
    if (sum >= n) break;
  }
  if (sum >= n)
    a0 = 1;
  else
    a0 = n / sum + (n % sum != 0);
  ans[1] = a0;
  m = n;
  m -= a0;
  for (i = 2; i <= k; i++) {
    ans[i] = ans[i - 1] + 1;
    m -= ans[i];
    sum = (ans[i] + 1 + ans[i] + k - i) * (k - i) / 2;
    if (m < sum) {
      ans[i] = n;
      break;
    }
    if (m > sum) {
      long long dx = m - sum;
      if (dx / (k - i + 1) > 2 * ans[i - 1] - ans[i]) {
        m = m - (2 * ans[i - 1] - ans[i]);
        sum = sum + (k - i) * (2 * ans[i - 1] - ans[i]);
        ans[i] = 2 * ans[i - 1];
      } else {
        m = m - dx / (k - i + 1);
        sum = sum + dx / (k - i + 1) * (k - i);
        ans[i] = ans[i] + dx / (k - i + 1);
      }
    } else if (m == sum) {
      for (j = i + 1; j <= k; j++) ans[j] = ans[j - 1] + 1;
      break;
    }
  }
  sum = 0;
  for (i = 1; i <= k; i++) sum += ans[i];
  if (sum != n)
    printf("NO\n");
  else {
    printf("YES\n");
    for (i = 1; i <= k; i++) printf("%lld ", ans[i]);
    printf("\n");
  }
}
