n, k = [int(s) for s in input().split()]
if n < k * (k + 1) / 2:
    print("NO")
    exit()
if ((k == 3 and n == 8) or n == 2 and k < 3):
    print("NO")
    exit()
print("YES")
if k == 1:
    print(n)
elif (k == 2):
    print((n + 2) // 3, n - (n + 2) // 3)
else:
    i = 1
    while i * k + k * (k - 1) // 2 <= n:
        i += 1
    i -= 1
    last = i + k - 1
    last_ = last - 1
    sum = i * (k - 1) + (k - 2) * (k - 1) // 2
    last = n - sum
    if last_ * 2 >= last:
        for q in range(i, last_ + 1):
            print(q, end = " ")
        print(last)
        exit()
    last_ += 1
    last -= 1
    for q in range(i, i + k - 2):
        print(q, end=" ")
    print(last_, end=' ')
    print(last)
