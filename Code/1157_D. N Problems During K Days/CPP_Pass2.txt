#include <bits/stdc++.h>
using namespace std;
int next() {
  int x;
  cin >> x;
  return x;
}
int main() {
  int n = next();
  int k = next();
  vector<long long> a(k);
  for (int i = 0; i < k; i++) a[i] = i + 1;
  long long sum = 0;
  for (auto i : a) sum += i;
  if (sum > n) {
    cout << "NO\n";
    return 0;
  }
  long long d = (n - sum) / k;
  for (int i = 0; i < k; i++) a[i] += d;
  sum += k * d;
  for (int i = k - 1; sum < n; i--) {
    sum++;
    a[i]++;
  }
  if (k > 1 && a[0] == 1 && a[1] == 3) {
    if (k <= 3) {
      cout << "NO\n";
      return 0;
    }
    a.back()++;
    a[1]--;
  }
  cout << "YES\n";
  for (auto aa : a) cout << aa << " ";
  cout << "\n";
}
