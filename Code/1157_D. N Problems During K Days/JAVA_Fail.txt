import java.util.Scanner;

public class D_555 {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        long n = in.nextLong();
        int k = in.nextInt();

        //impossible cases
        //During each day he has to solve at least one problem and one more than the previous day
        long min = k * (k + 1) /2;
        if (n < min) {
            System.out.println("NO");
            System.exit(0);
        }

        long[] a = new long[k];
        //base

        long h = (n - min) / k;
        a[0] = h + 1;
        for (int i = 1; i < k; i++) {
            a[i] = a[i - 1] + 1;
        }
        long sum = min + k * h;
        outer: while(sum < n) {
            for (int i = 0; i < k; i++) {
                if (sum + (k - i) <= n && (2* a[i - 1] > a[i])) {
                    while(i < k && sum < n) {
                        a[i++]++;
                        sum++;
                    }
                    continue  outer;
                }
            }
            System.out.println("NO");
            System.exit(0);
        }

        //System.out.println(sum);


        System.out.println("YES");
        for (int i = 0; i < a.length; i++) {
            System.out.print(a[i] + " ");
        }
        System.out.println();
    }
}
