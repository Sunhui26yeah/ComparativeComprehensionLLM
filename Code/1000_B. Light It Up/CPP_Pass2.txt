#include <bits/stdc++.h>
using namespace std;
int main() {
  ios::sync_with_stdio(0);
  int n, m;
  cin >> n >> m;
  vector<int> a(n + 1);
  for (int i = 1; i < n + 1; ++i) cin >> a[i];
  a.push_back(m);
  ;
  vector<int> os(n + 2), es(n + 2);
  int odd = 0;
  for (int i = n; i >= 0; --i) {
    if (i % 2 == 1) {
      es[i] = es[i + 1] + a[i + 1] - a[i];
      os[i] = os[i + 1];
    } else {
      os[i] = os[i + 1] + a[i + 1] - a[i];
      es[i] = es[i + 1];
    }
  };
  ;
  int total = 0, mtotal = os[0];
  for (int i = 0; i < n + 1; ++i) {
    if (i % 2 == 0) total += a[i + 1] - a[i];
    if (a[i + 1] - a[i] == 1) continue;
    if (i % 2 == 0) {
      ;
      int t = total - 1 + es[i + 1];
      ;
      mtotal = max(t, mtotal);
    } else {
      ;
      int t = total - 1 + a[i + 1] - a[i] + es[i + 1];
      ;
      mtotal = max(t, mtotal);
    }
  }
  cout << mtotal << endl;
  return 0;
}
