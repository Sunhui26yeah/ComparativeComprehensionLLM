def test(L,n):
    M=[0]*n
    f=0
    for i in range(n-1):
        if f==0:
            k=L[i]
            f=1
        if L[i]>L[i+1]:
            M[L[i]-1]=L[i+1]
        else:
            M[L[i]-1]=k
            f=0
        if i==n-2:
            if f==1:
                M[L[n-1]-1]=k
            else:
                M[L[n-1]-1]=L[n-1]
    return M==L
    
def next(L,n):
    for i in range(n-2,-1,-1):
        M=L[i:]
        M.sort()
        M.reverse()
        if L[i:]!=M:
            m=n
            s=L[i]
            for j in L[i+1:]:
                if j>s and j<m:
                    m=j
            M.remove(m)
            M.remove(s)
            L[i:]=[m,s]+M
            break

n,k=map(int,input().split(' '))
i=1

L=[i for i in range(1,n+1)]
while i<4:
    next(L,n)
    if test(L,n):
        i+=1
for i in L:
    print(i,end=' ')
