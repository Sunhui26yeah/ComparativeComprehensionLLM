#include <bits/stdc++.h>
using namespace std;
int N, K;
long long G[55];
bool vis[55];
int main() {
  G[0] = 1;
  for (int i = 1; i <= 20; i++) G[i] = G[i - 1] * i;
  cin >> N >> K;
  for (int i = 1; i <= N; i++)
    for (int j = 1; j <= N; j++)
      if (!vis[j])
        if (K > G[N - i])
          K -= G[N - i];
        else {
          printf("%d ", j);
          vis[j] = 1;
          break;
        }
}
