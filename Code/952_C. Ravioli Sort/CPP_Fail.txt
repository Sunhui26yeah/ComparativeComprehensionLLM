#include <bits/stdc++.h>
using namespace std;
int main() {
  int n;
  cin >> n;
  vector<int> a(n);
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }
  bool flag = 1;
  for (int i = 1; i < n; i++) {
    if (a[i] < a[i - 1]) {
      flag = 0;
      break;
    }
  }
  if (flag) {
    cout << "YES";
    return 0;
  }
  flag = 1;
  for (int i = 1; i < n; i++) {
    if (a[i] > a[i - 1]) {
      flag = 0;
      break;
    }
  }
  if (flag) {
    cout << "YES";
    return 0;
  }
  cout << "NO";
  return 0;
}
