#include <bits/stdc++.h>
using namespace std;
int main() {
  int n;
  bool oka = false;
  cin >> n;
  int a[n];
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }
  for (int i = 1; i < n; i++) {
    if (abs(a[i] - a[i - 1]) > 1) {
      oka = true;
      break;
    }
  }
  if (oka)
    cout << "NO" << endl;
  else
    cout << "YES" << endl;
}
