#include <bits/stdc++.h>
using namespace std;
int fail[1000000 + 10];
string s;
int n;
void fail_func() {
  int index = 0;
  for (int i = 1; i < n;) {
    if (s[i] == s[index]) {
      fail[i] = index + 1;
      index++;
      i++;
    } else {
      if (index != 0) {
        index = fail[index - 1];
      } else {
        fail[i] = 0;
        i++;
      }
    }
  }
}
int main() {
  cin >> s;
  n = s.size();
  fail_func();
  if (fail[n - 1] == 0) puts("Just a legend");
  int ans = fail[n - 1];
  for (int i = 0; i < n - 1; i++) {
    if (fail[i] == ans) {
      for (int j = 0; j < ans; j++) cout << s[j];
      cout << endl;
      return 0;
    }
  }
  puts("Just a legend");
  return 0;
}
