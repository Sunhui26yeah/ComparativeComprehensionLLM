#include <bits/stdc++.h>
using namespace std;
void solve() {
  int n;
  cin >> n;
  int a[n];
  int zero = 0;
  map<int, int> mp;
  for (int i = 0; i < n; i++) {
    cin >> a[i];
    mp[a[i]]++;
    if (a[i] == 0) zero++;
  }
  if (mp.size() == 1) {
    if (zero == 2) {
      cout << "2" << endl;
      return;
    } else {
      cout << "0" << endl;
      return;
    }
  }
  if (zero == 0) {
    cout << "0" << endl;
    return;
  } else if (zero == 1) {
    sort(a, a + n);
    int ans = 0;
    for (int i = 1; i < n; i++) {
      if (a[i] - a[i - 1] > 1) {
        ans = a[i - 1] + 1;
        break;
      }
    }
    if (ans == 0)
      cout << a[n - 1] + 1 << endl;
    else
      cout << ans << endl;
    return;
  } else {
    sort(a, a + n);
    int x = 0, ans = 0;
    for (int i = 1; i < n; i++) {
      if (a[i] - a[i - 1] > 1) {
        x = a[i - 1];
        ans = a[i - 1] + 1;
        break;
      }
    }
    if (ans == 0) {
      ans = a[n - 1] + 1;
      x = n - 1;
    }
    for (int i = 1; i <= x; i++) {
      if (mp[i] == 1) {
        ans = ans + i;
        break;
      }
    }
    cout << ans << endl;
    return;
  }
}
int main() {
  int t;
  cin >> t;
  while (t--) {
    solve();
  }
}
