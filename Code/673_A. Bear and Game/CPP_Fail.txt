#include <bits/stdc++.h>
using namespace std;
int main() {
  long long n;
  cin >> n;
  long long l[n + 1];
  l[0] = 0;
  for (int i = 1; i <= n; i++) {
    cin >> l[i];
  }
  for (int i = 1; i <= n; i++) {
    if (l[i] - l[i - 1] >= 15) {
      cout << l[i - 1] + 15;
      return 0;
    }
  }
  cout << 90;
}
