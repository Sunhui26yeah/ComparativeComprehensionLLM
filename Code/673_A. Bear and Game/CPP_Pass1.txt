#include <bits/stdc++.h>
using namespace std;
int n;
bool m[100];
int main() {
  scanf("%d", &n);
  for (int i = 1; i <= n; i++) {
    int t;
    scanf("%d", &t);
    m[t] = true;
  }
  for (int i = 0, j = 0; i <= 90; i++, j++) {
    if (j > 15) {
      printf("%d", i - 1);
      return 0;
    }
    if (m[i]) j = 0;
  }
  printf("%d", 90);
}
