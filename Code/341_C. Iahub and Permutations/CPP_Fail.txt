#include <bits/stdc++.h>
using namespace std;
const int MOD = 1e9 + 7;
int n, m, k, d[2001], a[2001];
int f[2001][2001];
int main() {
  cin >> n;
  m = n;
  for (int i = 1; i <= n; i++) {
    cin >> a[i];
    if (a[i] != -1) k++, d[i] = 1;
  }
  for (int i = 1; i <= n; i++)
    if (a[i] != -1 && a[a[i]] == -1) m--;
  n -= k;
  m -= k;
  f[0][0] = 1;
  for (int i = 1; i <= n; i++) f[i][0] = f[i - 1][0] * i % MOD;
  for (int i = 1; i <= n; i++)
    for (int j = 1; j <= m; j++)
      f[i][j] = (f[i][j - 1] - f[i - 1][j - 1] + MOD) % MOD;
  cout << f[n][m];
}
