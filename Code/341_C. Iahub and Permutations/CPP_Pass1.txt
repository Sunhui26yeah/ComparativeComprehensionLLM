#include <bits/stdc++.h>
const int MAX_N = 2005;
const int MOD = 1e9 + 7;
int m, n, k;
int a[MAX_N];
bool mem[MAX_N];
long long dp[MAX_N][MAX_N];
int main() {
  scanf("%d", &m);
  for (int i = 1; i <= m; i++) {
    scanf("%d", &a[i]);
    if (a[i] > 0) {
      mem[a[i]] = true;
    }
  }
  for (int i = 1; i <= m; i++) {
    if (a[i] == -1) {
      n++;
      k += not mem[i];
    }
  }
  dp[0][0] = 1;
  for (int i = 1; i <= n; i++) {
    dp[i][0] = dp[i - 1][0] * i % MOD;
    dp[i][1] = dp[i - 1][0] * (i - 1) % MOD;
    for (int j = 2; j <= i and j <= k; j++) {
      dp[i][j] = (i - j) * dp[i - 1][j - 1] + (j - 1) * dp[i - 1][j - 2];
      dp[i][j] %= MOD;
    }
  }
  printf("%d\n", (int)dp[n][k]);
}
